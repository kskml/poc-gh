import os, requests, uuid
from datetime import datetime
from langfuse import Langfuse

# --- CONFIGURATION ---
GITHUB_TOKEN = os.getenv("GITHUB_TOKEN", "ghp_your_token_here")
GITHUB_ORG = os.getenv("GITHUB_ORG", "your-org-name")

langfuse = Langfuse(
    public_key=os.getenv("LANGFUSE_PUBLIC_KEY", "pk-lf-xxx"),
    secret_key=os.getenv("LANGFUSE_SECRET_KEY", "sk-lf-xxx"),
    host=os.getenv("LANGFUSE_HOST", "https://cloud.langfuse.com") # or http://localhost:3000
)

def fetch_and_ingest():
    url = f"https://api.github.com/orgs/{GITHUB_ORG}/copilot/usage"
    headers = {"Authorization": f"Bearer {GITHUB_TOKEN}", "Accept": "application/json"}
    
    response = requests.get(url, headers=headers)
    response.raise_for_status()
    metrics = response.json()

    for day_data in metrics:
        day = day_data.get("day")
        trace_id = f"copilot-{day}-{uuid.uuid4().hex[:8]}"
        
        total_sug = day_data.get("total_suggestions_count", 0)
        total_acc = day_data.get("total_acceptances_count", 0)
        acc_rate = (total_acc / total_sug * 100) if total_sug > 0 else 0

        trace = langfuse.trace(
            id=trace_id,
            name="copilot_daily_metrics",
            metadata={"date": day, "org": GITHUB_ORG},
            tags=["github-copilot", "intellij"]
        )

        # Log overall stats
        trace.generation(
            name="daily_acceptance_rate",
            model="copilot-aggregate",
            input={"suggestions": total_sug},
            output={"acceptances": total_acc, "acceptance_rate_pct": round(acc_rate, 2)},
            usage={"prompt_tokens": total_sug, "completion_tokens": total_acc}
        )

        # Log per-language/IDE breakdown (filters for IntelliJ if desired)
        for item in day_data.get("breakdown", []):
            if item.get("editor") == "intellij": # Focus on IntelliJ
                trace.span(
                    name=f"usage_{item.get('language')}_intellij",
                    output=item
                )

    langfuse.flush()
    print(f"[✅] Ingested {len(metrics)} days of Copilot metrics to Langfuse.")

if __name__ == "__main__":
    fetch_and_ingest()
