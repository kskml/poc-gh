Enterprise Release Diff Analyzer (v7 - Final)
1. GitHub API: Fetches raw diff, falls back to JSON for >1MB, fetches full files if context is too small.
2. Tree-Sitter: Fault-tolerant Java parsing.
3. Enhanced Java Intent Engine: Deep behavioral extraction (handles JPA, Streams, Optionals).
4. Liquibase Parser: State-machine that translates XML changesets into DB schema summaries.
5. Strict Exclusions: Tests, Docs, POMs, Build files are completely ignored.
"""

EXCLUDED_PATHS = [
    r'^[^/]+/src/test/', r'^[^/]+/src/integration-test/', r'(^|/)tests?(/|\.java$)',
    r'IT\.java$', r'Tests\.java$', r'Test\.java$', r'^README', r'^[^/]+/docs/',
    r'\.md$', r'pom\.xml$', r'build\.gradle(\.kts)?$', r'settings\.gradle(\.kts)?$',
    r'\.json$', r'^\.github/', r'^LICENSE'
]

def should_exclude(file_path: str) -> bool:
    return any(re.search(p, file_path.replace('\\', '/'), re.IGNORECASE) for p in EXCLUDED_PATHS)

def fetch_github_diff(owner: str, repo: str, base: str, head: str, token: str = None) -> str:
    url = f"https://api.github.com/repos/{owner}/{repo}/compare/{base}...{head}"
    headers = {"User-Agent": "Release-Analyzer"}
    if token: headers["Authorization"] = f"token {token}"
        
    print(f"🔄 Fetching diff: {owner}/{repo} ({base} -> {head})...")
    
    # Attempt 1: Raw diff
    h_diff = {**headers, "Accept": "application/vnd.github.v3.diff"}
    res = requests.get(url, headers=h_diff)
    if res.status_code == 200 and len(res.text.strip()) > 10:
        print("✅ Fetched raw diff."); return res.text
        
    # Attempt 2: JSON Payload Fallback (For >1MB diffs)
    print("⚠️ Diff too large, using JSON fallback...")
    h_json = {**headers, "Accept": "application/vnd.github.v3+json"}
    res = requests.get(url, headers=h_json)
    if res.status_code != 200: raise Exception(f"API Error: {res.status_code}")
    data = res.json()
    if data.get("status") == "identical": return ""
    
    diff_text = ""
    for f in data.get("files", []):
        if f.get("patch"): diff_text += f"diff --git a/{f['filename']} b/{f['filename']}\n{f['patch']}\n"
    print("✅ Reconstructed diff."); return diff_text

def fetch_full_file(owner: str, repo: str, path: str, ref: str, token: str) -> Optional[str]:
    url = f"https://api.github.com/repos/{owner}/{repo}/contents/{path}"
    headers = {"User-Agent": "Release-Analyzer"}
    if token: headers["Authorization"] = f"token {token}"
    res = requests.get(url, headers=headers, params={"ref": ref})
    if res.status_code == 200: return base64.b64decode(res.json()["content"]).decode('utf-8', errors='ignore')
    return None

def extract_xml_attr(line: str, attr: str) -> Optional[str]:
    match = re.search(rf'{attr}\s*=\s*"([^"]+)"', line, re.IGNORECASE)
    return match.group(1) if match else None

def parse_liquibase_changes(lines: List[str]) -> str:
    """
    Parses Liquibase XML using a state machine to understand context
    (e.g., knowing which table a column belongs to).
    """
    intents = []
    current_table = None
    current_constraint = None
    
    noise_words = {'xmlns', 'xmlns:xsi', 'xsi:schemalocation', 'objectquotingstrategy', 'logicalfilepath'}
    
    for line in lines:
        line = line.strip()
        if not line or line.startswith('<?') or line.startswith('--'): continue
        
        # Extract base tags regardless of state
        tag_match = re.match(r'<(\w+)', line)
        tag = tag_match.group(1).lower() if tag_match else ""
        
        # Skip standard XML noise
        if tag in ['databasechangelog', 'changelog', 'property', 'preconditions', 'comment']: continue
        if any(nw in tag for nw in noise_words): continue
        
        # State tracking for context
        if 'createtable' in tag:
            current_table = extract_xml_attr(line, 'tableName')
            if current_table: intents.append(f"Created table: `{current_table}`")
            
        elif 'droptable' in tag:
            t = extract_xml_attr(line, 'tableName') or extract_xml_attr(line, 'cascadeConstraints')
            intents.append(f"Dropped table: `{t}`" if t else "Dropped table")
            
        elif 'renamecolumn' in tag:
            old = extract_xml_attr(line, 'oldColumnName')
            new = extract_xml_attr(line, 'newColumnName')
            intents.append(f"Renamed column `{old}` to `{new}` in `{current_table}`")
            
        elif 'renameTable' in tag.lower():
            old = extract_xml_attr(line, 'oldTableName')
            new = extract_xml_attr(line, 'newTableName')
            intents.append(f"Renamed table `{old}` to `{new}`")
            
        elif 'addcolumn' in tag:
            current_table = extract_xml_attr(line, 'tableName') or current_table
            intents.append(f"Altered table `{current_table}` (adding columns)")
            
        elif 'dropcolumn' in tag:
            current_table = extract_xml_attr(line, 'tableName') or current_table
            col = extract_xml_attr(line, 'columnName')
            intents.append(f"Dropped column `{col}` from `{current_table}`")
            
        elif 'adduniqueconstraint' in tag or 'addForeignKeyConstraint' in tag.lower():
            constraint_name = extract_xml_attr(line, 'constraintName')
            ref_table = extract_xml_attr(line, 'referencedTableName')
            action = "Added unique constraint" if 'unique' in tag else f"Added foreign key to `{ref_table}`"
            intents.append(f"{action} `{constraint_name}` on `{current_table}`")
            
        elif 'dropForeignKeyConstraint' in tag.lower() or 'dropUniqueConstraint' in tag.lower():
            c = extract_xml_attr(line, 'constraintName')
            intents.append(f"Dropped constraint `{c}`")
            
        elif 'createindex' in tag:
            idx = extract_xml_attr(line, 'indexName')
            t = extract_xml_attr(line, 'tableName') or current_table
            intents.append(f"Created index `{idx}` on `{t}`")
            
        elif 'dropindex' in tag:
            idx = extract_xml_attr(line, 'indexName')
            intents.append(f"Dropped index `{idx}`")
            
        elif 'insert' in tag:
            t = extract_xml_attr(line, 'tableName')
            intents.append(f"Seeded data into `{t}`")
            
        elif 'column' in tag and current_table:
            # Extract column specifics if inside an addColumn block
            col_name = extract_xml_attr(line, 'name')
            col_type = extract_xml_attr(line, 'type')
            if col_name and col_type:
                # Avoid duplicating if we already said "adding columns"
                if not any(f"Added column `{col_name}`" in i for i in intents):
                    intents.append(f"Added column `{col_name}` ({col_type})")
            elif col_name:
                constraints = []
                if 'true' == extract_xml_attr(line, 'nullable'): constraints.append("NULL")
                if 'true' == extract_xml_attr(line, 'unique'): constraints.append("UNIQUE")
                if constraints:
                    intents.append(f"Column `{col_name}` constraints: {', '.join(constraints)}")

    # Deduplicate and clean up
    seen = set()
    final = []
    for i in intents:
        if i not in seen: seen.add(i); final.append(i)
    return "; ".join(final) if final else "Modified XML changelog structure"

def is_liquibase_file(lines: List[str]) -> bool:
    return any('<changeSet' in l or 'liquibase' in l.lower() for l in lines[:20])

INTENT_PATTERNS = [
    (r'throw\s+new\s+(\w+Exception)', r'Threw \1'),
    (r'if\s*\([^)]*==\s*null[^)]*\)', 'Added null-check guard'),
    (r'if\s*\([^)]*!=\s*null[^)]*\)', 'Added non-null validation'),
    (r'Objects\.requireNonNull', 'Enforced non-null via Objects'),
    (r'log\.(debug|info|warn|error)', 'Added diagnostic logging'),
    (r'@Valid(?:ated)?\b', 'Enforced request validation'),
    (r'@Transactional\s*\(', 'Applied transaction boundary'),
    (r'@Cacheable|@CacheEvict|@CachePut', 'Integrated caching layer'),
    (r'@ExceptionHandler', 'Mapped exception to HTTP response'),
    (r'ResponseEntity\.(ok|status|notFound|noContent|badRequest)', 'Wrapped response in HTTP entity'),
    (r'\.orElseThrow\(\s*(?:\(\)\s*->)?\s*new', 'Enforced Optional presence with custom exception'),
    (r'\.orElse\(null\)', 'Fallback to null if empty'),
    (r'\.ifPresent\(', 'Executed side-effect if value exists'),
    (r'\.map\(\s*\w+::', 'Transformed object using method reference'),
    (r'\.stream\(\)\.collect\(', 'Aggregated stream to collection'),
    (r'\.filter\(', 'Filtered collection elements'),
    (r'\.sorted\(', 'Sorted collection elements'),
    (r'\.persist\(', 'JPA: Persisted new entity (managed)'),
    (r'\.merge\(', 'JPA: Merged detached entity'),
    (r'new\s+\w+Exception\(', 'Instantiated custom exception'),
    (r'public\s+record\s+\w+', 'Defined immutable Java 17 record'),
    (r'sealed\s+(class|interface)', 'Defined Java 17 sealed hierarchy'),
    (r'hikari|maximum-pool|minimum-idle', 'Tuned database connection pool'),
    (r'Collections\.emptyList|Collections\.singleton', 'Returned immutable empty/singleton collection'),
]

NOISE_WORDS = {'public', 'private', 'protected', 'static', 'final', 'void', 'return', 'new', 'if', 'else', 'this', 'super'}

def extract_java_intent(lines: List[str]) -> str:
    intents = []
    for line in lines:
        stripped = line.strip()
        if not stripped or stripped.startswith('//') or stripped.startswith('*'): continue
            
        # 1. High-level regex patterns
        matched = False
        for pattern, intent in INTENT_PATTERNS:
            match = re.search(pattern, stripped, re.IGNORECASE)
            if match:
                final_intent = intent
                if isinstance(intent, str) and match.lastindex and match.lastindex >= 1:
                    # Handle dynamic replacement like r'Threw \1'
                    try: final_intent = intent.replace(r'\1', match.group(1))
                    except: pass
                if final_intent not in intents: intents.append(final_intent)
                matched = True
                break
        if matched: continue

        # 2. Deep extraction: Strip syntax to find Verb + Object
        clean = stripped
        clean = re.sub(r'\b(public|private|protected|static|final|abstract|synchronized|native|transient|volatile|void|int|long|double|float|boolean|char|String|List|Set|Map|Optional|var)\b', '', clean, flags=re.IGNORECASE).strip()
        clean = re.sub(r'@\w+(\([^)]*\))?\s*', '', clean).strip()
        clean = re.sub(r'<[^>]+>', '', clean) # Generics
        if not clean or clean in ['{', '}', ';', '();', '{}']: continue

        # Handle return statements elegantly
        if clean.startswith('return '):
            obj = clean.replace('return ', '').replace(';', '').strip()
            if len(obj) > 3 and obj not in NOISE_WORDS: intents.append(f"Returned processed result: {obj[:40]}")
            continue

        # Handle method invocations (e.g., repository.save(entity))
        inv_match = re.search(r'(\w+(?:\.\w+)*)\s*\(([^)]{0,50})\)', clean)
        if inv_match:
            method = inv_match.group(1).split('.')[-1] # Get last part (e.g., 'save' from 'this.repository.save')
            args = inv_match.group(2).strip()
            
            # Filter out standard loops/conditionals
            if method not in ['for', 'while', 'switch', 'if', 'else', 'catch']:
                if len(args) > 30: args = args[:30] + "..."
                action = f"Invoked `{method}`"
                if args: action += f" on `{args}`"
                intents.append(action)
            continue

        # Handle variable assignments (e.g., this.status = Status.ACTIVE)
        assign_match = re.match(r'(?:this\.)?(\w+)\s*=\s*([^;]+);?', clean)
        if assign_match:
            field_name, value = assign_match.group(1), assign_match.group(2).strip()
            if field_name not in ['i', 'j', 'k', 'count', 'result', 'str', 'tmp', 'e'] and len(value) > 2:
                if len(value) > 40: value = value[:40] + "..."
                intents.append(f"Assigned state: `{field_name}` = `{value}`")
            continue
            
        # Final Fallback (Highly refined)
        if len(stripped) > 15:
            words = [w for w in stripped.replace(';', '').replace('{', '').replace('}', '').split() 
                     if len(w) > 2 and w.lower() not in NOISE_WORDS][:4]
            if words: intents.append(f"Executed logic: {' '.join(words)}")

    seen = set()
    unique = [i for i in intents if not (i in seen or seen.add(i))]
    return "; ".join(unique) if unique else "Refactored internal execution block"

def get_text(node: ts.Node, src: bytes) -> str:
    return src[node.start_byte:node.end_byte].decode('utf8', errors='ignore')

def find_context(node: ts.Node, src: bytes) -> Tuple[Optional[str], Optional[str]]:
    cur = node.parent
    m, c = None, None
    while cur:
        if cur.type == 'method_declaration' and not m:
            n = cur.child_by_field_name('name')
            if n: m = get_text(n, src)
        if cur.type in ('class_declaration', 'interface_declaration') and not c:
            n = cur.child_by_field_name('name')
            if n: c = get_text(n, src)
        if m and c: break
        cur = cur.parent
    return m, c

def get_class_meta(root: ts.Node, src: bytes, target: str) -> Dict:
    meta = {'component': None, 'endpoints': [], 'package': None}
    for child in root.children:
        if child.type == 'package_declaration':
            n = child.child_by_field_name('name')
            if n: meta['package'] = get_text(n, src)
    for child in root.children:
        if child.type in ('class_declaration', 'interface_declaration', 'enum_declaration', 'record_declaration'):
            n = child.child_by_field_name('name')
            if n and get_text(n, src) == target:
                for ann in child.children:
                    if ann.type == 'marker_annotation':
                        txt = get_text(ann, src)
                        if 'RestController' in txt: meta['component'] = 'RestController'
                        elif 'Controller' in txt: meta['component'] = 'Controller'
                        elif 'Service' in txt: meta['component'] = 'Service'
                        elif 'Repository' in txt: meta['component'] = 'Repository'
                        elif 'Configuration' in txt: meta['component'] = 'Configuration'
                body = next((c for c in child.children if 'body' in c.type), None)
                if body: 
                    txt = get_text(body, src)
                    for a, m in {'@GetMapping': 'GET', '@PostMapping': 'POST', '@PutMapping': 'PUT', '@DeleteMapping': 'DELETE'}.items():
                        match = re.search(rf'{a}\s*\(\s*(?:value\s*=\s*)?["\']([^"\']+)["\']', txt)
                        if a in txt: meta['endpoints'].append(f"{m} {match.group(1) if match else '/'}")
                break
    return meta

def process_java_full_file(added: List[str], hunks: List[Dict], full_src: str) -> Dict:
    src = full_src.encode('utf8')
    tree = PARSER.parse(src)
    root = tree.root_node
    lines = full_src.split('\n')
    res = {'package': None, 'classes': {}, 'needs_fallback': False}
    intents = defaultdict(lambda: {'added': []})
    
    cur_new = 0
    for hunk in hunks:
        cur_new = hunk['new_start']
        for l_type, content in hunk['lines']:
            if l_type == 'added':
                b_off = sum(len(lines[i]) + 1 for i in range(cur_new - 1)) if cur_new <= len(lines) else 0
                node = root.named_descendant_for_byte_range(b_off, b_off + len(content.encode('utf8')))
                if node:
                    m, c = find_context(node, src)
                    if m and c: intents[(c, m)]['added'].append(content)
                    else: res['needs_fallback'] = True
                cur_new += 1
            elif l_type == 'context': cur_new += 1

    cls_map = {}
    for (cn, mn), ld in intents.items():
        if cn not in cls_map:
            meta = get_class_meta(root, src, cn)
            cls_map[cn] = {'type': 'class', 'component': meta['component'], 'endpoints': meta['endpoints'], 'methods': {}}
            if meta['package'] and not res['package']: res['package'] = meta['package']
        cls_map[cn]['methods'][mn] = f"Added: {extract_java_intent(ld['added'])}"
    res['classes'] = cls_map
    return res

def process_java_diff_ctx(added: List[str], removed: List[str], ctx: List[str]) -> Dict:
    code = "\n".join(ctx + added)
    src = code.encode('utf8')
    tree = PARSER.parse(src)
    root = tree.root_node
    res = {'package': None, 'classes': {}, 'needs_fallback': False}
    
    for l in ctx:
        m = re.match(r'^\s*package\s+([\w.]+)', l)
        if m: res['package'] = m.group(1); break

    cls_node = next((c for c in root.children if c.type in ('class_declaration', 'interface_declaration', 'enum_declaration', 'record_declaration')), None)
    if not cls_node: res['needs_fallback'] = True; return res

    n = cls_node.child_by_field_name('name')
    cls_name = get_text(n, src) if n else "Unknown"
    comp = None
    for c in cls_node.children:
        if c.type == 'marker_annotation':
            t = get_text(c, src)
            if 'RestController' in t: comp = 'RestController'
            elif 'Controller' in t: comp = 'Controller'
            elif 'Service' in t: comp = 'Service'
            elif 'Repository' in t: comp = 'Repository'

    eps = []
    txt = "\n".join(added)
    for a, m in {'@GetMapping': 'GET', '@PostMapping': 'POST', '@PutMapping': 'PUT', '@DeleteMapping': 'DELETE'}.items():
        match = re.search(rf'{a}\s*\(\s*(?:value\s*=\s*)?["\']([^"\']+)["\']', txt)
        if a in txt: eps.append(f"{m} {match.group(1) if match else '/'}")

    m_intents = {}
    for line in added + removed:
        idx = code.find(line)
        if idx == -1: continue
        node = root.named_descendant_for_byte_range(idx, idx + len(line.encode('utf8')))
        if not node: continue
        mn, _ = find_context(node, src)
        if mn:
            if mn not in m_intents: m_intents[mn] = {'added': [], 'removed': []}
            if line in added: m_intents[mn]['added'].append(line)
            else: m_intents[mn]['removed'].append(line)
        else: res['needs_fallback'] = True

    if res['needs_fallback'] or not m_intents: res['needs_fallback'] = True
    else:
        methods_map = {}
        for mn, ld in m_intents.items():
            act = "Modified" if ld['added'] and ld['removed'] else ("Added" if ld['added'] else "Removed")
            methods_map[mn] = f"{act}: {extract_java_intent(ld['added'] + ld['removed'])}"
        res['classes'][cls_name] = {'type': cls_node.type.replace('_declaration', ''), 'component': comp, 'endpoints': eps, 'methods': methods_map}
    return res

def parse_diff(diff_text: str) -> List[Dict]:
    sections = re.split(r'^diff --git ', diff_text, flags=re.MULTILINE)
    if sections and not sections[0].strip(): sections = sections[1:]
    res = []
    for sec in sections:
        lines = sec.split('\n')
        m = re.match(r'a/(.+?)\s+b/(.+)', lines[0])
        if not m or should_exclude(m.group(2)): continue
        status = 'Modified'
        if 'new file' in sec[:200]: status = 'Added'
        elif 'deleted file' in sec[:200]: status = 'Deleted'
        hunks = []
        cur = None
        for l in lines:
            hm = re.match(r'^@@ -(\d+)(?:,(\d+))? \+(\d+)(?:,(\d+))? @@', l)
            if hm:
                if cur: hunks.append(cur)
                cur = {'new_start': int(hm.group(3)), 'lines': []}
            elif cur:
                if l.startswith('+'): cur['lines'].append(('added', l[1:]))
                elif l.startswith('-'): cur['lines'].append(('removed', l[1:]))
                elif l.startswith(' '): cur['lines'].append(('context', l[1:]))
        if cur: hunks.append(cur)
        if any(h['lines'] for h in hunks): res.append({'path': m.group(2), 'status': status, 'hunks': hunks})
    return res

def generate_summary(diff_text: str, owner: str, repo: str, head: str, token: str) -> str:
    files = parse_diff(diff_text)
    if not files: return "No relevant changes found."
    release = ReleaseSummary()
    stats = {'java': 0, 'api': 0, 'db': 0, 'fetched': 0}
    
    for f in files:
        p = f['path']
        is_java = p.endswith('.java')
        is_xml = p.endswith('.xml')
        added = [c for h in f['hunks'] for t, c in h['lines'] if t == 'added']
        removed = [c for h in f['hunks'] for t, c in h['lines'] if t == 'removed']
        ctx = [c for h in f['hunks'] for t, c in h['lines'] if t == 'context']
        
        f_sum = FileSummary(path=p, status=f['status'])
        
        # Route to correct parser
        if is_java and (added or removed):
            stats['java'] += 1
            ast = process_java_diff_ctx(added, removed, ctx)
            if ast.get('needs_fallback'):
                print(f"   ⚠️ Context too small for {p}, fetching full file...")
                full_src = fetch_full_file(owner, repo, p, head, token)
                if full_src:
                    stats['fetched'] += 1
                    f_sum.used_full_file_fetch = True
                    ast = process_java_full_file(added, f['hunks'], full_src)
            
            f_sum.package = ast.get('package')
            for cn, cd in ast.get('classes', {}).items():
                cc = ClassChange(name=cn, type=cd.get('type','class'), component=cd.get('component'), endpoints=cd.get('endpoints',[]))
                for mn, intent in cd.get('methods', {}).items():
                    cc.methods.append(MethodChange(name=mn, intent=intent))
                f_sum.classes.append(cc)
                stats['api'] += len(cc.endpoints)
                
        elif is_xml and (added or removed):
            if is_liquibase_file(added + ctx):
                stats['db'] += 1
                f_sum.db_intent = parse_liquibase_changes(added + removed)
            else:
                f_sum.config_intent = "Updated XML configuration"
                
        elif any(p.endswith(x) for x in ['.yml', '.yaml', '.properties']) and (added or removed):
            f_sum.config_intent = extract_java_intent(added + removed) # Re-use logic for config keys
            
        release.files.append(f_sum)

    release.overview = f"Java: {stats['java']} | APIs impacted: {stats['api']} | DB Changesets: {stats['db']} | Full-files fetched: {stats['fetched']}"
    return format_llm(release)

def format_llm(release: ReleaseSummary) -> str:
    out = [f"# Release Change Summary\n**Overview:** {release.overview}\n"]
    for f in release.files:
        if not f.classes and not f.config_intent and not f.db_intent: continue
        hdr = f"## `{f.path}` ({f.status})"
        if f.package: hdr += f" | Pkg: `{f.package}`"
        if f.used_full_file_fetch: hdr += " *[Full-File Context]*"
        out.append(hdr)
        
        for c in f.classes:
            comp = f" [{c.component}]" if c.component else ""
            out.append(f"- **{c.type} {c.name}**{comp}")
            if c.endpoints: out.append(f"  - *Endpoints involved:* {', '.join(c.endpoints)}")
            for m in c.methods: out.append(f"  - **{m.name}**: {m.intent}")
            
        if f.db_intent: out.append(f"- **Database Schema:** {f.db_intent}")
        if f.config_intent: out.append(f"- **Config Change:** {f.config_intent}")
        out.append("")
    return "\n".join(out)

# ============================================================================
# CLI ENTRY POINT
# ============================================================================

def main():
    parser = argparse.ArgumentParser(description="Enterprise Release Diff Analyzer")
    parser.add_argument("owner", help="GitHub Owner")
    parser.add_argument("repo", help="GitHub Repo")
    parser.add_argument("base_tag", help="Base Tag (e.g., v1.0)")
    parser.add_argument("head_tag", help="Head Tag (e.g., v1.1)")
    parser.add_argument("-o", "--output", default="release_summary.md")
    args = parser.parse_args()
    
    token = os.getenv("GITHUB_TOKEN")
    if not token: print("⚠️ Warning: GITHUB_TOKEN not set. Rate limits may apply.\n")

    try:
        diff = fetch_github_diff(args.owner, args.repo, args.base_tag, args.head_tag, token)
        if not diff.strip(): print("Nothing to process."); sys.exit(0)
        
        print("🧠 Analyzing changes (Tree-Sitter + Liquibase Engine)...")
        summary = generate_summary(diff, args.owner, args.repo, args.head_tag, token)
        
        with open(args.output, 'w', encoding='utf-8') as file:
            file.write(summary)
        print(f"\n✅ Saved to: {args.output}")
    except Exception as e:
        print(f"\n❌ Error: {e}"); sys.exit(1)

if __name__ == "__main__":
    main()
