"""
Hyper-Optimized Diff Summarizer for LLM Context
Focus: Zero raw code in output. 100% behavioral/intent summaries.
Uses Tree-Sitter to locate context, heuristics to summarize intent.
"""

import re
import tree_sitter_java as tsjava
import tree_sitter as ts
from typing import Dict, List, Optional
from dataclasses import dataclass, field

# Initialize Tree-sitter
JAVA_LANGUAGE = ts.Language(tsjava.language())
PARSER = ts.Parser(JAVA_LANGUAGE)

# ============================================================================
# ULTRA-COMPACT DATA STRUCTURES (No raw code stored here)
# ============================================================================

@dataclass
class MethodChange:
    name: str
    intent: str  # e.g., "Added null-check validation and logging"

@dataclass
class ClassChange:
    name: str
    type: str
    component: Optional[str] # e.g., "RestController"
    endpoints: List[str] = field(default_factory=list)
    methods: List[MethodChange] = field(default_factory=list)
    structural_change: str = "" # e.g., "Added record for immutable data"

@dataclass
class FileSummary:
    path: str
    status: str
    package: Optional[str] = None
    classes: List[ClassChange] = field(default_factory=list)
    config_intent: str = "" # For yml/properties files

@dataclass
class ReleaseSummary:
    overview: str = ""
    files: List[FileSummary] = field(default_factory=list)
    breaking_indicators: List[str] = field(default_factory=list)

# ============================================================================
# STEP 1: INTENT EXTRACTION ENGINE (The Magic)
# ============================================================================

INTENT_PATTERNS = [
    (r'throw\s+new\s+\w+', 'Throws explicit exception'),
    (r'if\s*\([^)]*==\s*null[^)]*\)', 'Added null-check guard'),
    (r'if\s*\([^)]*!=\s*null[^)]*\)', 'Added null-check guard'),
    (r'log\.(debug|info|warn|error|trace)', 'Added logging statement'),
    (r'@Valid(?:ated)?\b', 'Added request validation'),
    (r'@Transactional\s*\(', 'Added transaction boundary'),
    (r'@Cacheable|@CacheEvict', 'Added caching layer'),
    (r'ResponseEntity\.(ok|status|notFound|noContent)', 'Wrapped response in ResponseEntity'),
    (r'\.stream\(\)\.map\(', 'Added stream-based DTO mapping'),
    (r'\.orElseThrow\(', 'Replaced optional get with orElseThrow'),
    (r'\.orElse\(null\)', 'Fallback to null if empty'),
    (r'\.save\(|\.saveAll\(', 'Persisted entity to database'),
    (r'\.deleteBy|\.remove\(', 'Deleted entity from database'),
    (r'\.findById\(', 'Fetched entity by ID'),
    (r'\.findAll\(|\.findAllBy', 'Queried multiple entities'),
    (r'new\s+\w+DTO\(', 'Instantiated new Data Transfer Object'),
    (r'public\s+record\s+\w+', 'Defined immutable Java 17 record'),
    (r'sealed\s+(class|interface)', 'Defined Java 17 sealed type hierarchy'),
    (r'permits\s+[\w,\s]+', 'Restricted inheritance (permits)'),
    (r'hikari|maximum-pool|minimum-idle', 'Tuned database connection pool'),
    (r'spring\.datasource\.url', 'Modified database connection string'),
]

def summarize_code_intent(lines: List[str]) -> List[str]:
    """Translate raw code lines into human-readable behavioral intents."""
    intents = set()
    
    for line in lines:
        stripped = line.strip()
        if not stripped or stripped.startswith('//') or stripped.startswith('*'):
            continue
            
        for pattern, intent in INTENT_PATTERNS:
            if re.search(pattern, stripped, re.IGNORECASE):
                intents.add(intent)
                
    return list(intents)

def infer_diff_type(added: List[str], removed: List[str]) -> str:
    """Infer if this is an addition, modification, or removal block."""
    if not removed and added: return "Added"
    if not added and removed: return "Removed"
    return "Modified"

# ============================================================================
# STEP 2: CONTEXT-AWARE TREE-SITTER PARSER
# ============================================================================

def get_node_text(node: ts.Node, source_bytes: bytes) -> str:
    return source_bytes[node.start_byte:node.end_byte].decode('utf8', errors='ignore')

def find_enclosing_method_name(node: ts.Node, source_bytes: bytes) -> Optional[str]:
    """Walk up the tree from a changed line to find which method it belongs to."""
    current = node.parent
    while current:
        if current.type == 'method_declaration':
            name_node = current.child_by_field_name('name')
            if name_node:
                return get_node_text(name_node, source_bytes)
        if current.type in ('class_declaration', 'interface_declaration', 'enum_declaration'):
            break # Don't go outside the class
        current = current.parent
    return None

def extract_endpoints_from_annotations(text: str) -> List[str]:
    """Extract HTTP method and path without keeping raw annotations."""
    endpoints = []
    # Normalize multiline annotations to single line
    text = re.sub(r'\s+', ' ', text)
    
    mappings = {
        '@GetMapping': 'GET', '@PostMapping': 'POST', '@PutMapping': 'PUT',
        '@DeleteMapping': 'DELETE', '@PatchMapping': 'PATCH'
    }
    
    for ann, method in mappings.items():
        if ann in text:
            # Extract path
            match = re.search(rf'{ann}\s*\(\s*(?:value\s*=\s*)?["\']([^"\']+)["\']', text)
            path = match.group(1) if match else '/'
            endpoints.append(f"{method} {path}")
            
    return endpoints

def process_java_change(added: List[str], removed: List[str], context: List[str]) -> Dict:
    """
    Uses Tree-Sitter on the CONTEXT lines to figure out WHERE we are,
    then maps the ADDED/REMOVED intents to that location.
    """
    # Combine context to make a valid enough block for Tree-Sitter
    context_code = "\n".join(context + added) 
    source_bytes = context_code.encode('utf8')
    tree = PARSER.parse(source_bytes)
    
    result = {
        'package': None,
        'classes': []
    }
    
    # Extract package via simple regex (faster than AST for this)
    for line in context:
        pkg_match = re.match(r'^\s*package\s+([\w.]+)', line)
        if pkg_match:
            result['package'] = pkg_match.group(1)
            break

    # Find class and method context using Tree-Sitter
    root = tree.root_node
    
    # We only care about the first class/interface in this snippet
    class_node = None
    for child in root.children:
        if child.type in ('class_declaration', 'interface_declaration', 'enum_declaration', 'record_declaration'):
            class_node = child
            break
            
    if not class_node:
        # If no class found (e.g., top-level code or error), just return intents
        intents = summarize_code_intent(added + removed)
        if intents:
            result['classes'].append({'type': 'Unknown', 'name': 'Unknown', 'component': None, 'endpoints': [], 'methods': [{'name': 'Block', 'intent': ", ".join(intents)}]})
        return result

    # Extract Class Info
    name_node = class_node.child_by_field_name('name')
    class_name = get_node_text(name_node, source_bytes) if name_node else "Unknown"
    class_type = class_node.type.replace('_declaration', '')
    
    component = None
    for child in class_node.children:
        if child.type == 'marker_annotation':
            ann_text = get_node_text(child, source_bytes)
            if '@RestController' in ann_text: component = 'RestController'
            elif '@Controller' in ann_text: component = 'Controller'
            elif '@Service' in ann_text: component = 'Service'
            elif '@Repository' in ann_text: component = 'Repository'
            elif '@Configuration' in ann_text: component = 'Configuration'

    # Extract Endpoints from added text block
    endpoints = extract_endpoints_from_annotations("\n".join(added))
    
    # Map intents to specific methods
    method_intents = {}
    
    # To find methods, we look at added/removed lines in the context of the AST
    all_change_lines = added + removed
    
    for line in all_change_lines:
        # Find this line in the AST
        line_bytes = line.encode('utf8')
        # Rough search for the node containing this text
        start_idx = context_code.find(line)
        if start_idx == -1: continue
        
        node = root.named_descendant_for_byte_range(start_idx, start_idx + len(line_bytes))
        if not node: continue
        
        method_name = find_enclosing_method_name(node, source_bytes)
        if method_name:
            if method_name not in method_intents:
                method_intents[method_name] = set()
            # Tag if added or removed
            if line in added:
                method_intents[method_name].add("Added")
            else:
                method_intents[method_name].add("Removed")

    # Generate final method summaries
    methods_summary = []
    raw_intents = summarize_code_intent(all_change_lines)
    
    if method_intents:
        for m_name, tags in method_intents.items():
            action = " & ".join(tags)
            # Combine action with behavioral intents
            final_intent = f"{action}: {', '.join(raw_intents)}" if raw_intents else f"{action} logic"
            methods_summary.append({'name': m_name, 'intent': final_intent})
    elif raw_intents:
        # Happened outside a method (e.g., field injection)
        methods_summary.append({'name': 'Class-level', 'intent': ", ".join(raw_intents)})
    else:
        methods_summary.append({'name': 'Internal', 'intent': 'Updated implementation details'})

    result['classes'].append({
        'type': class_type,
        'name': class_name,
        'component': component,
        'endpoints': endpoints,
        'methods': methods_summary
    })
    
    return result

# ============================================================================
# STEP 3: CONFIG FILE PARSER (YML/Properties)
# ============================================================================

def summarize_config_change(added: List[str], removed: List[str]) -> str:
    """Convert config changes to plain English."""
    intents = summarize_code_intent(added + removed)
    if intents:
        return "; ".join(intents)
        
    # Fallback: just list the keys changed
    keys = []
    for line in added:
        line = line.strip()
        if ':' in line and not line.startswith('#'):
            keys.append(line.split(':')[0].strip())
    if keys:
        return f"Modified keys: {', '.join(keys[:5])}"
    return "Updated configuration"

# ============================================================================
# STEP 4: MAIN PIPELINE
# ============================================================================

def parse_diff_text(diff_text: str) -> List[Dict]:
    """Minimal diff parser to extract hunks efficiently."""
    diff_text = diff_text.replace('\r\n', '\n')
    file_sections = re.split(r'^diff --git ', diff_text, flags=re.MULTILINE)
    if file_sections and not file_sections[0].strip(): file_sections = file_sections[1:]
    
    results = []
    for section in file_sections:
        lines = section.split('\n')
        match = re.match(r'a/(.+?)\s+b/(.+)', lines[0])
        if not match: continue
        
        status = 'Modified'
        if 'new file' in section[:200]: status = 'Added'
        elif 'deleted file' in section[:200]: status = 'Deleted'
        
        hunks = []
        cur_hunk = None
        for line in lines:
            if re.match(r'^@@', line):
                if cur_hunk: hunks.append(cur_hunk)
                cur_hunk = {'added': [], 'removed': [], 'context': []}
            elif cur_hunk:
                if line.startswith('+'): cur_hunk['added'].append(line[1:])
                elif line.startswith('-'): cur_hunk['removed'].append(line[1:])
                elif line.startswith(' '): cur_hunk['context'].append(line[1:])
        if cur_hunk: hunks.append(cur_hunk)
        
        results.append({'path': match.group(2), 'status': status, 'hunks': hunks})
    return results

def generate_llm_summary(diff_text: str) -> str:
    """Main function: Converts massive diffs into ultra-compact semantic summaries."""
    files = parse_diff_text(diff_text)
    release = ReleaseSummary()
    
    total_java, total_eps = 0, 0
    all_endpoints = []
    
    for f in files:
        path = f['path']
        is_java = path.endswith('.java')
        is_config = any(path.endswith(x) for x in ['.yml', '.yaml', '.properties'])
        
        added = [l for h in f['hunks'] for l in h['added']]
        removed = [l for h in f['hunks'] for l in h['removed']]
        context = [l for h in f['hunks'] for l in h['context']]
        
        f_sum = FileSummary(path=path, status=f['status'])
        
        if is_java and (added or removed):
            total_java += 1
            ast_data = process_java_change(added, removed, context)
            f_sum.package = ast_data['package']
            
            for c in ast_data['classes']:
                c_change = ClassChange(name=c['name'], type=c['type'], component=c['component'], endpoints=c['endpoints'])
                for m in c['methods']:
                    c_change.methods.append(MethodChange(name=m['name'], intent=m['intent']))
                f_sum.classes.append(c_change)
                
                all_endpoints.extend(c['endpoints'])
                
        elif is_config and (added or removed):
            f_sum.config_intent = summarize_config_change(added, removed)
            
        elif added or removed:
            f_sum.config_intent = f"Updated {len(added)} lines"
            
        release.files.append(f_sum)

    total_eps = len(all_endpoints)
    release.overview = f"Files changed: {len(files)} | Java: {total_java} | APIs impacted: {total_eps}"
    
    # Check for breaking indicators
    for f in release.files:
        if f.status == 'Deleted' and any(c.component == 'RestController' for c in f.classes):
             release.breaking_indicators.append(f"Deleted Controller: {f.path}")

    return format_for_llm(release)

def format_for_llm(release: ReleaseSummary) -> str:
    """Formats the data structure into pure natural language LLM context."""
    output = [f"# Release Change Summary\n**Overview:** {release.overview}\n"]
    
    if release.breaking_indicators:
        output.append("⚠️ **POTENTIAL BREAKING CHANGES:**\n" + "\n".join(f"- {b}" for b in release.breaking_indicators) + "\n")
    
    for f in release.files:
        # Skip completely empty files
        if not f.classes and not f.config_intent: continue
        
        header = f"## `{f.path}` ({f.status})"
        if f.package: header += f" | Pkg: `{f.package}`"
        output.append(header)
        
        for c in f.classes:
            comp_str = f" [{c.component}]" if c.component else ""
            output.append(f"- **{c.type} {c.name}**{comp_str}")
            
            if c.endpoints:
                output.append(f"  - *Endpoints involved:* {', '.join(c.endpoints)}")
                
            for m in c.methods:
                output.append(f"  - **{m.name}**: {m.intent}")
                
        if f.config_intent:
            output.append(f"- **Config Change:** {f.config_intent}")
            
        output.append("") # Spacing
        
    return "\n".join(output)
