import re
import json

# ==========================================
# 1. CUSTOM GITHUB DIFF PARSER
# ==========================================
def _parse_github_diff_raw(diff_text: str):
    current_file = None
    added_lines_buffer = []
    file_pattern = re.compile(r'^diff --git a/(.*) b/(.*)$')
    
    for line in diff_text.split('\n'):
        file_match = file_pattern.match(line)
        if file_match:
            if current_file is not None:
                yield current_file, added_lines_buffer
            current_file = file_match.group(2)
            added_lines_buffer = []
            continue
            
        if current_file and line.startswith('+') and not line.startswith('+++'):
            added_lines_buffer.append(line[1:]) 
            
    if current_file is not None:
        yield current_file, added_lines_buffer

# ==========================================
# 2. STRUCTURAL REGEX MAPPER (Replaces AST)
# ==========================================
def _clean_body_lines(lines: list) -> str:
    """Flattens logic to save LLM tokens."""
    cleaned = []
    for line in lines:
        stripped = line.strip()
        if stripped and stripped != '{' and stripped != '}':
            cleaned.append(stripped)
    return "\n    ".join(cleaned)

def _filter_imports(import_lines: list) -> list:
    """Drops standard Java boilerplate imports to save tokens."""
    filtered = []
    skip_prefixes = ('java.lang.', 'java.util.', 'java.io.', 'java.net.', 'lombok.')
    for imp in import_lines:
        if not imp.startswith(skip_prefixes):
            filtered.append(imp)
    return filtered

def _process_java_file(code_lines: list) -> dict:
    """
    Uses advanced Regex to group raw diff lines into Classes/Methods.
    Keeps 100% of implementation logic. Immune to partial diff syntax errors.
    """
    imports = [line.strip() for line in code_lines if line.strip().startswith('import ')]
    filtered_imports = _filter_imports(imports)
    
    # Regex patterns to identify Java structures (handles inline Spring annotations)
    class_pattern = re.compile(r'^\s*(?:public|private|protected)?\s*(?:abstract|static)?\s*(class|interface|enum|record)\s+(\w+)')
    method_pattern = re.compile(
        r'^\s*(?:(@\w+(?:\([^)]*\))?)\s*)*' # Captures inline annotations like @Override
        r'(?:public|private|protected)?\s*' 
        r'(?:static|final|abstract|synchronized|default)?\s*' 
        r'([\w<>\[\],\s\?]+?)\s+' # Return type
        r'(\w+)\s*' # Method name
        r'\(([^)]*)\)\s*' # Parameters
        r'(?:throws\s+[\w,\s]+)?\s*' # Throws clause
        r'\{?\s*$' # Opening brace
    )
    field_pattern = re.compile(
        r'^\s*(?:(@\w+(?:\([^)]*\))?)\s*)*' # Captures @Autowired etc
        r'(?:private|protected|public)?\s*' 
        r'([\w<>\[\]]+)\s+' # Type
        r'(\w+)\s*[;=]' # Field name
    )
    ann_pattern = re.compile(r'^\s*(@\w+(?:\([^)]*\))?)\s*$') # Standalone annotations
    
    classes_data = []
    current_class = {"name": "Root", "type": "File", "annotations": [], "methods": []}
    current_method = None
    ann_buffer = []
    
    for line in code_lines:
        stripped = line.strip()
        if not stripped: continue
        
        # 1. Buffer standalone annotations (e.g., lines with JUST @Service)
        if ann_pattern.match(stripped):
            ann_buffer.append(stripped)
            continue
            
        # 2. Detect Class/Interface
        c_match = class_pattern.match(stripped)
        if c_match:
            if current_class["methods"]:
                classes_data.append(current_class)
            current_class = {
                "name": c_match.group(2), 
                "type": c_match.group(1), 
                "annotations": list(ann_buffer), 
                "methods": []
            }
            ann_buffer = []
            current_method = None
            continue
            
        # 3. Detect Method Signature
        m_match = method_pattern.match(stripped)
        if m_match:
            signature = f"{m_match.group(1).strip()} {m_match.group(2)}({m_match.group(3).strip()})"
            current_method = {
                "signature": signature, 
                "annotations": list(ann_buffer), 
                "logic_body": []
            }
            current_class["methods"].append(current_method)
            ann_buffer = []
            continue
            
        # 4. Detect Fields (only outside of methods to avoid confusing variables with fields)
        f_match = field_pattern.match(stripped)
        if f_match and not current_method:
            signature = f"FIELD: {f_match.group(1)} {f_match.group(2)}"
            current_class["methods"].append({
                "signature": signature, 
                "annotations": list(ann_buffer), 
                "logic_body": [stripped]
            })
            ann_buffer = []
            continue
            
        # 5. Capture implementation logic inside the method
        if current_method:
            if stripped != '{' and stripped != '}':
                current_method["logic_body"].append(stripped)
                
    # Don't forget the last class in the file
    if current_class["methods"]:
        classes_data.append(current_class)
        
    # Finalize string formatting
    for cls in classes_data:
        for method in cls["methods"]:
            if isinstance(method["logic_body"], list):
                method["logic_body"] = "\n    ".join(method["logic_body"])
                
    return {"imports": filtered_imports, "classes": classes_data}

# ==========================================
# 3. MAIN EXECUTION & JSON EXPORT
# ==========================================
def extract_structured_code(diff_text: str, output_json_path: str) -> None:
    final_data = []

    for file_path, added_lines in _parse_github_diff_raw(diff_text):
        if not file_path.endswith('.java'): 
            continue
        if not added_lines: continue

        # Process using the stable Regex mapper
        file_data = _process_java_file(added_lines)
        file_data["file"] = file_path
        
        final_data.append(file_data)

    with open(output_json_path, 'w', encoding='utf-8') as f:
        json.dump(final_data, f, indent=4)
        
    print(f"\nSuccess! Extraction saved to: {output_json_path}")
