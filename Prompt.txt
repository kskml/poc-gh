import re
import json
from javalang import parse, tree
from javalang.parser import JavaSyntaxError

# ==========================================
# 1. CUSTOM GITHUB DIFF PARSER
# ==========================================
def _parse_github_diff_raw(diff_text: str):
    current_file = None
    added_lines_buffer = []
    file_pattern = re.compile(r'^diff --git a/(.*) b/(.*)$')
    
    for line in diff_text.split('\n'):
        file_match = file_pattern.match(line)
        if file_match:
            if current_file is not None:
                yield current_file, added_lines_buffer
            current_file = file_match.group(2)
            added_lines_buffer = []
            continue
            
        if current_file and line.startswith('+') and not line.startswith('+++'):
            added_lines_buffer.append(line[1:]) 
            
    if current_file is not None:
        yield current_file, added_lines_buffer

# ==========================================
# 2. TOKEN-OPTIMIZED AST PROCESSOR
# ==========================================
def _clean_body_lines(lines: list) -> str:
    """Flattens a method body to save LLM tokens (removes isolated braces/empty lines)."""
    cleaned = []
    for line in lines:
        stripped = line.strip()
        # Skip empty lines and lines that are ONLY a curly brace
        if stripped and stripped != '{' and stripped != '}':
            cleaned.append(stripped)
    return "\n    ".join(cleaned)

def _filter_imports(import_lines: list) -> list:
    """Drops standard Java boilerplate imports to save tokens."""
    filtered = []
    skip_prefixes = ('java.lang.', 'java.util.', 'java.io.', 'java.net.', 'lombok.')
    for imp in import_lines:
        if not imp.startswith(skip_prefixes):
            filtered.append(imp)
    return filtered

def _process_java_file_ast(code_lines: list) -> dict:
    """
    Uses AST to map the file into Classes/Methods, but keeps raw logic inside.
    """
    code_str = "\n".join(code_lines)
    is_partial = False
    
    # 1. Extract Imports via Regex (AST is bad at partial imports)
    imports = [line.strip() for line in code_lines if line.strip().startswith('import ')]
    filtered_imports = _filter_imports(imports)
    
    # 2. Parse to AST
    try:
        ast_tree = parse(code_str)
    except JavaSyntaxError:
        try:
            wrapped_code = f"public class DiffWrapper {{ {code_str} }}"
            ast_tree = parse(wrapped_code)
            is_partial = True
        except Exception:
            # Fallback if completely unparsable
            return {"imports": filtered_imports, "unmapped_code": _clean_body_lines(code_lines)}

    classes_data = []
    
    # Find top-level types
    types = ast_tree.filter(tree.TypeDeclaration)
    if is_partial:
        # If wrapped, extract the inner body of the dummy class
        dummy = ast_tree.filter(tree.ClassDeclaration)
        if dummy: types = dummy[0].body

    for t in types:
        class_info = {
            "name": t.name,
            "type": t.__class__.__name__.replace('Declaration', ''),
            "annotations": [a.name for a in getattr(t, 'annotations', [])],
            "methods": []
        }
        
        # Walk class body for methods and fields
        for member in getattr(t, 'body', []):
            if isinstance(member, tree.MethodDeclaration):
                # Find where this method is in the raw code lines to grab its body
                start_line = member.position.line - 1 if member.position else 0
                # Estimate end line (next method or end of class)
                end_line = len(code_lines) 
                for next_member in getattr(t, 'body', []):
                    if next_member.position and next_member.position.line > member.position.line:
                        end_line = next_member.position.line - 1
                        break
                
                body_raw = code_lines[start_line:end_line]
                
                # Extract signature from AST
                params = []
                for p in getattr(member, 'parameters', []):
                    param_anns = " ".join([a.name for a in getattr(p, 'annotations', [])])
                    p_str = f"{param_anns} {p.type.name} {p.name}".strip()
                    params.append(p_str)
                
                ret_type = member.return_type.name if member.return_type else "void"
                
                class_info["methods"].append({
                    "signature": f"{ret_type} {member.name}({', '.join(params)})",
                    "annotations": [a.name for a in member.annotations],
                    "logic_body": _clean_body_lines(body_raw) # Keep logic, flatten tokens
                })
                
            elif isinstance(member, tree.FieldDeclaration) and member.annotations:
                 # Only capture Fields if they have annotations (e.g. @Autowired)
                 f_type = member.type.name
                 f_name = member.declarators[0].name if member.declarators else "unknown"
                 class_info["methods"].append({
                    "signature": f"FIELD: {f_type} {f_name}",
                    "annotations": [a.name for a in member.annotations],
                    "logic_body": None
                 })

        classes_data.append(class_info)

    return {
        "imports": filtered_imports,
        "classes": classes_data
    }

# ==========================================
# 3. MAIN EXECUTION & JSON EXPORT
# ==========================================
def extract_token_optimized_ast(diff_text: str, output_json_path: str) -> None:
    final_data = []

    for file_path, added_lines in _parse_github_diff_raw(diff_text):
        if not file_path.endswith('.java'): 
            continue # Basic constraint: Focus on Java for Spring Boot
        if not added_lines: continue

        file_data = _process_java_file_ast(added_lines)
        file_data["file"] = file_path
        
        final_data.append(file_data)

    with open(output_json_path, 'w', encoding='utf-8') as f:
        json.dump(final_data, f, indent=4)
        
    print(f"Token-optimized AST extraction saved to: {output_json_path}")
