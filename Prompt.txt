# --- GITHUB DIFF PRE-PROCESSOR ---
def _sanitize_github_diff(diff_text: str) -> list:
    """
    Fixes common GitHub API diff formatting issues that cause unidiff to crash.
    Splits the diff into individual file blocks for safe parsing.
    """
    # 1. Remove GitHub binary file diff lines completely (they have no hunks)
    clean_text = re.sub(r'^Binary files .* differ\s*$', '', diff_text, flags=re.MULTILINE)
    
    # 2. Split into individual file diffs
    # GitHub diffs always start with "diff --git a/... b/..."
    file_blocks = re.split(r'^(?=diff --git)', clean_text, flags=re.MULTILINE)
    
    sanitized_blocks = []
    for block in file_blocks:
        if not block.strip():
            continue
            
        lines = block.split('\n')
        fixed_lines = []
        in_hunk = False
        
        for line in lines:
            # Detect start of a hunk
            if line.startswith('@@'):
                in_hunk = True
                fixed_lines.append(line)
                continue
                
            # Detect new file block (resets hunk state)
            if line.startswith('diff --git'):
                in_hunk = False
                fixed_lines.append(line)
                continue
                
            # Fix lines INSIDE hunks
            if in_hunk:
                # Valid diff lines start with +, -, \, or ' ' (space)
                if line.startswith(('+', '-', '\\', ' ')):
                    fixed_lines.append(line)
                elif line == '':
                    # Empty line in a hunk MUST be a space (context line) for unidiff
                    fixed_lines.append(' ')
                else:
                    # GitHub API bug: missing leading space on context line. Add it.
                    fixed_lines.append(' ' + line)
            else:
                # Headers, metadata, etc. Leave alone
                fixed_lines.append(line)
                
        sanitized_blocks.append("\n".join(fixed_lines))
        
    return sanitized_blocks

# --- SAFE PARSER ---
def _get_patched_files_safely(diff_text: str):
    """Yields PatchedFile objects, skipping files that fail to parse."""
    sanitized_blocks = _sanitize_github_diff(diff_text)
    
    for block in sanitized_blocks:
        try:
            # Parse the sanitized block
            patch_set = PatchSet.from_string(block)
            for patched_file in patch_set:
                yield patched_file
        except Exception as e:
            # If a specific file still fails, skip it gracefully
            print(f"[Warning] Skipping malformed diff block: {str(e)[:80]}...")
            continue

# --- MAIN EXECUTION FUNCTION (UPDATED) ---
def extract_java_ast_architecture(diff_text: str) -> str:
    """
    Extracts Added and Modified code blocks from raw GitHub Diff text.
    Resistant to GitHub API formatting quirks.
    """
    final_blueprint = []

    # Use the safe generator instead of raw PatchSet
    for patched_file in _get_patched_files_safely(diff_text):
        if not patched_file.path.endswith('.java'): continue
        if any(skip in patched_file.path for skip in ['Test.java', 'Tests.java']): continue

        # Extract ONLY added lines (captures both Additions and Modifications)
        added_lines = [line.value for hunk in patched_file for line in hunk if line.is_added]
        if not added_lines: continue

        file_context = f"// File: {patched_file.path}"
        
        # Try AST first
        ast_nodes = _parse_java_snippet(added_lines)
        
        if ast_nodes:
            extracted_lines = _walk_ast_nodes(ast_nodes)
            if extracted_lines:
                final_blueprint.append(file_context)
                final_blueprint.extend(extracted_lines)
        else:
            # Fallback to Regex for fragmented modifications
            fallback_lines = _extract_via_regex(added_lines)
            if fallback_lines:
                final_blueprint.append(f"{file_context} (Fragmented Modification)")
                final_blueprint.extend(fallback_lines)

    return "\n".join(final_blueprint)
