import os

# Initialize Tree-sitter
JAVA_LANGUAGE = ts.Language(tsjava.language())
PARSER = ts.Parser(JAVA_LANGUAGE)

# ============================================================================
# DATA STRUCTURES
# ============================================================================

@dataclass
class MethodChange:
    name: str
    intent: str 

@dataclass
class ClassChange:
    name: str
    type: str
    component: Optional[str]
    endpoints: List[str] = field(default_factory=list)
    methods: List[MethodChange] = field(default_factory=list)

@dataclass
class FileSummary:
    path: str
    status: str
    package: Optional[str] = None
    classes: List[ClassChange] = field(default_factory=list)
    config_intent: str = ""
    db_intent: str = ""

@dataclass
class ReleaseSummary:
    overview: str = ""
    files: List[FileSummary] = field(default_factory=list)

# ============================================================================
# STEP 1: GITHUB API & EXCLUSIONS
# ============================================================================

EXCLUDED_PATHS = [
    r'^[^/]+/src/test/', r'^[^/]+/src/integration-test/', r'(^|/)tests?(/|\.java$)',
    r'IT\.java$', r'Tests\.java$', r'Test\.java$', r'^README', r'^[^/]+/docs/',
    r'\.md$', r'pom\.xml$', r'build\.gradle(\.kts)?$', r'settings\.gradle(\.kts)?$',
    r'\.json$', r'^\.github/', r'^LICENSE'
]

def should_exclude(p: str) -> bool:
    return any(re.search(pat, p.replace('\\', '/'), re.IGNORECASE) for pat in EXCLUDED_PATHS)

def fetch_diff(owner: str, repo: str, base: str, head: str, token: str = None) -> str:
    url = f"https://api.github.com/repos/{owner}/{repo}/compare/{base}...{head}"
    headers = {"User-Agent": "Release-Analyzer"}
    if token: headers["Authorization"] = f"token {token}"
    print(f"🔄 Fetching diff: {owner}/{repo} ({base} -> {head})...")
    
    res = requests.get(url, headers={**headers, "Accept": "application/vnd.github.v3.diff"})
    if res.status_code == 200 and len(res.text.strip()) > 10: 
        print("✅ Fetched raw diff."); return res.text
        
    print("⚠️ Diff too large, reconstructing via JSON...")
    res = requests.get(url, headers={**headers, "Accept": "application/vnd.github.v3+json"})
    if res.status_code != 200: raise Exception(f"API Error: {res.status_code}")
    data = res.json()
    if data.get("status") == "identical": return ""
    print("✅ Reconstructed diff.")
    return "".join(f"diff --git a/{f['filename']} b/{f['filename']}\n{f.get('patch','')}\n" for f in data.get("files",[]) if f.get("patch"))

# ============================================================================
# STEP 2: LIQUIBASE XML STATE PARSER
# ============================================================================

def get_xml_attr(line: str, attr: str) -> Optional[str]:
    m = re.search(rf'{attr}\s*=\s*"([^"]+)"', line, re.IGNORECASE)
    return m.group(1) if m else None

def parse_liquibase(lines: List[str]) -> str:
    intents, cur_table = [], None
    for l in lines:
        l = l.strip()
        if not l or l.startswith('<?') or l.startswith('--'): continue
        if '</changeSet>' in l: cur_table = None # Reset state per changeset
        
        tag = re.match(r'<(\w+)', l)
        t = tag.group(1).lower() if tag else ""
        if t in ['databasechangelog', 'changelog', 'property']: continue
        
        if 'createtable' in t:
            cur_table = get_xml_attr(l, 'tableName')
            if cur_table: intents.append(f"Created table `{cur_table}`")
        elif 'droptable' in t: 
            intents.append(f"Dropped table `{get_xml_attr(l, 'tableName')}`")
        elif 'renamecolumn' in t: 
            intents.append(f"Renamed column `{get_xml_attr(l, 'oldColumnName')}` to `{get_xml_attr(l, 'newColumnName')}`")
        elif 'addcolumn' in t:
            cur_table = get_xml_attr(l, 'tableName') or cur_table
            intents.append(f"Altered table `{cur_table}` (added cols)")
        elif 'dropcolumn' in t:
            cur_table = get_xml_attr(l, 'tableName') or cur_table
            intents.append(f"Dropped column `{get_xml_attr(l, 'columnName')}` from `{cur_table}`")
        elif 'column' in t and cur_table:
            cn, ct = get_xml_attr(l, 'name'), get_xml_attr(l, 'type')
            if cn and ct and not any(f"`{cn}`" in i for i in intents):
                intents.append(f"Added column `{cn}` ({ct})")
        elif 'adduniqueconstraint' in t: 
            intents.append(f"Added unique constraint `{get_xml_attr(l, 'constraintName')}`")
        elif 'addForeignKeyConstraint' in t.lower(): 
            intents.append(f"Added FK to `{get_xml_attr(l, 'referencedTableName')}`")
        elif 'createindex' in t: 
            intents.append(f"Created index `{get_xml_attr(l, 'indexName')}`")
        elif 'insert' in t: 
            intents.append(f"Seeded data in `{get_xml_attr(l, 'tableName')}`")
            
    return "; ".join(list(dict.fromkeys(intents))) if intents else "Modified XML changelog"

# ============================================================================
# STEP 3: ENHANCED JAVA INTENT ENGINE
# ============================================================================

INTENT_PATTERNS = [
    (r'throw\s+new\s+(\w+)', r'Threw \1'), 
    (r'if\s*\([^)]*==\s*null', 'Added null-check'), 
    (r'Objects\.requireNonNull', 'Enforced non-null'), 
    (r'log\.(debug|info|warn|error)', 'Added logging'),
    (r'@Valid(?:ated)?\b', 'Enforced validation'), 
    (r'@Transactional\b', 'Applied transaction'),
    (r'@Cacheable|@CacheEvict', 'Integrated caching'), 
    (r'ResponseEntity\.(ok|status|notFound)', 'Wrapped in HTTP entity'),
    (r'\.orElseThrow\(', 'Enforced Optional presence'), 
    (r'\.map\(\s*\w+::', 'Transformed via method reference'),
    (r'\.stream\(\).collect', 'Aggregated stream'), 
    (r'\.persist\(', 'JPA persisted entity'),
    (r'sealed\s+(class|interface)', 'Defined sealed hierarchy'), 
    (r'public\s+record\s+\w+', 'Defined Java 17 record'),
]

def extract_intent(lines: List[str]) -> str:
    intents = []
    for l in lines:
        s = l.strip()
        if not s or s.startswith('//'): continue
        matched = False
        for pat, intent in INTENT_PATTERNS:
            m = re.search(pat, s, re.IGNORECASE)
            if m:
                final = intent.replace(r'\1', m.group(1)) if m.lastindex else intent
                if final not in intents: intents.append(final)
                matched = True; break
        if matched: continue
        
        # Deep extraction fallback
        clean = re.sub(r'\b(?:public|private|protected|static|final|void|int|String|List|Optional|var)\b', '', s, flags=re.IGNORECASE).strip()
        clean = re.sub(r'@\w+(\([^)]*\))?\s*', '', clean).strip()
        clean = re.sub(r'<[^>]+>', '', clean)
        
        if clean.startswith('return ') and len(clean) > 10: 
            intents.append(f"Returned: {clean[7:40]}"); continue
        inv = re.search(r'(\w+)\s*\(([^)]{0,40})\)', clean)
        if inv and inv.group(1) not in ['for','while','switch','if','else']:
            arg_str = f" on `{inv.group(2)[:20]}`" if inv.group(2).strip() else ""
            intents.append(f"Invoked `{inv.group(1)}`{arg_str}")
            
    return "; ".join(list(dict.fromkeys(intents))) if intents else "Refactored execution block"

# ============================================================================
# STEP 4: TREE-SITTER + ZERO-FETCH FALLBACK ENGINE (v9 Optimized)
# ============================================================================

def get_text(node: ts.Node, src: bytes) -> str:
    return src[node.start_byte:node.end_byte].decode('utf8', errors='ignore')

def find_ts_context(node: ts.Node, src: bytes) -> Tuple[Optional[str], Optional[str]]:
    cur, m, c = node.parent, None, None
    while cur:
        if cur.type == 'method_declaration' and not m:
            n = cur.child_by_field_name('name')
            if n: m = get_text(n, src)
        if cur.type in ('class_declaration', 'interface_declaration', 'enum_declaration', 'record_declaration') and not c:
            n = cur.child_by_field_name('name')
            if n: c = get_text(n, src)
        if m and c: break
        cur = cur.parent
    return m, c

# Strict regex patterns for backtracking
METHOD_RE = re.compile(r'^\s*(?:@\w+\s+)*(?:(?:public|private|protected)\s+)?(?:(?:static|final|abstract)\s+)*(?:<[^>]+>\s+)?[\w<>?\[\]\.]+\s+(\w+)\s*\(')
CLASS_RE = re.compile(r'^\s*(?:@\w+\s+)*(?:(?:public|private|protected)\s+)?(?:(?:abstract|static|final|sealed|non-sealed)\s+)*(class|interface|enum|record)\s+(\w+)')

def process_java_zero_fetch(raw_lines: List[str]) -> Dict:
    # IMPROVEMENT: Strip diff headers for Tree-Sitter to avoid parsing 'diff --git' as Java
    ts_code_lines = [l[1:] for l in raw_lines if l and l[0] in '+- ']
    ts_code = "\n".join(ts_code_lines)
    src = ts_code.encode('utf8')
    tree = PARSER.parse(src)
    root = tree.root_node
    
    res = {'package': None, 'classes': {}}
    
    # Get package
    for l in ts_code_lines:
        m = re.match(r'^\s*package\s+([\w.]+)', l)
        if m: res['package'] = m.group(1); break

    for idx, line in enumerate(raw_lines):
        if not line or line[0] not in '+-': continue
        content = line[1:]
        m_name, c_name, c_type = None, None, "class"
        
        # ATTEMPT 1: Tree-Sitter AST Walk
        code_idx = ts_code.find(content)
        if code_idx != -1:
            node = root.named_descendant_for_byte_range(code_idx, code_idx + len(content.encode('utf8')))
            if node: 
                ts_m, ts_c = find_ts_context(node, src)
                if ts_m: m_name = ts_m
                if ts_c: c_name = ts_c
            
        # ATTEMPT 2: Regex Backtrack (Zero API calls)
        if not m_name or not c_name:
            for i in range(idx - 1, max(-1, idx - 150), -1):
                if not m_name:
                    l_clean = raw_lines[i][1:] if raw_lines[i] and raw_lines[i][0] in '+- ' else raw_lines[i]
                    m = METHOD_RE.match(l_clean.strip())
                    if m: m_name = m.group(1)
            for i in range(idx - 1, max(-1, idx - 500), -1):
                if not c_name:
                    l_clean = raw_lines[i][1:] if raw_lines[i] and raw_lines[i][0] in '+- ' else raw_lines[i]
                    c = CLASS_RE.match(l_clean.strip())
                    if c: 
                        c_type = c.group(1) # Capture if it's interface/record/class
                        c_name = c.group(2)

        # ATTEMPT 3: Path Heuristic Inference
        if not c_name and raw_lines[0].startswith('diff --git'):
            path_match = re.search(r'b/(?:.*?/)?(\w+)\.java', raw_lines[0])
            if path_match: c_name = path_match.group(1)
            if not m_name: m_name = "internal_logic"

        if c_name and m_name:
            if c_name not in res['classes']:
                comp = None
                for i in range(max(0, idx - 50), idx):
                    if '@RestController' in raw_lines[i]: comp = 'RestController'
                    elif '@Controller' in raw_lines[i]: comp = 'Controller'
                    elif '@Service' in raw_lines[i]: comp = 'Service'
                    elif '@Repository' in raw_lines[i]: comp = 'Repository'
                    elif '@Configuration' in raw_lines[i]: comp = 'Configuration'
                # Store the type (class/interface/record) found by regex, default to ts if not found
                res['classes'][c_name] = {'type': c_type, 'component': comp, 'endpoints': [], 'methods': {}}
                
            if m_name not in res['classes'][c_name]['methods']:
                 res['classes'][c_name]['methods'][m_name] = {'added': [], 'removed': []}
                 
            res['classes'][c_name]['methods'][m_name]['added' if line[0] == '+' else 'removed'].append(content)

    # Format intents and endpoints
    for c_name, cd in res['classes'].items():
        eps = []
        for mn, ld in cd['methods'].items():
            txt = "\n".join(ld['added'])
            for a, m in {'@GetMapping': 'GET', '@PostMapping': 'POST', '@PutMapping': 'PUT', '@DeleteMapping': 'DELETE'}.items():
                match = re.search(rf'{a}\s*\(\s*(?:value\s*=\s*)?["\']([^"\']+)["\']', txt)
                if a in txt: eps.append(f"{m} {match.group(1) if match else '/'}")
            act = "Modified" if ld['added'] and ld['removed'] else ("Added" if ld['added'] else "Removed")
            cd['methods'][mn] = f"{act}: {extract_intent(ld['added'] + ld['removed'])}"
        cd['endpoints'] = eps
    return res

def parse_diff(diff_text: str) -> List[Dict]:
    sections = re.split(r'^diff --git ', diff_text, flags=re.MULTILINE)
    if sections and not sections[0].strip(): sections = sections[1:]
    res = []
    for sec in sections:
        lines = sec.split('\n')
        m = re.match(r'a/(.+?)\s+b/(.+)', lines[0])
        if not m or should_exclude(m.group(2)): continue
        status = 'Added' if 'new file' in sec[:200] else ('Deleted' if 'deleted file' in sec[:200] else 'Modified')
        hunks, cur = [], None
        for l in lines:
            if re.match(r'^@@', l):
                if cur: hunks.append(cur)
                hm = re.match(r'^@@ -\d+(?:,\d+)? \+(\d+)', l)
                cur = {'new_start': int(hm.group(1)), 'lines': []} if hm else {'new_start': 0, 'lines': []}
            elif cur:
                if l.startswith('+'): cur['lines'].append(('added', l[1:]))
                elif l.startswith('-'): cur['lines'].append(('removed', l[1:]))
                elif l.startswith(' '): cur['lines'].append(('context', l[1:]))
        if cur: hunks.append(cur)
        if any(h['lines'] for h in hunks): res.append({'path': m.group(2), 'status': status, 'hunks': hunks, 'raw_lines': lines})
    return res

def generate_summary(diff_text: str) -> str:
    files = parse_diff(diff_text)
    if not files: return "No relevant changes found."
    release = ReleaseSummary()
    stats = {'java': 0, 'api': 0, 'db': 0}
    
    for f in files:
        p, is_java, is_xml = f['path'], f['path'].endswith('.java'), f['path'].endswith('.xml')
        added = [c for h in f['hunks'] for t, c in h['lines'] if t == 'added']
        removed = [c for h in f['hunks'] for t, c in h['lines'] if t == 'removed']
        f_sum = FileSummary(path=p, status=f['status'])
        
        if is_java and (added or removed):
            stats['java'] += 1
            ast = process_java_zero_fetch(f['raw_lines'])
            f_sum.package = ast.get('package')
            for cn, cd in ast.get('classes', {}).items():
                cc = ClassChange(name=cn, type=cd.get('type','class'), component=cd.get('component'), endpoints=cd.get('endpoints',[]))
                for mn, intent in cd.get('methods', {}).items():
                    cc.methods.append(MethodChange(name=mn, intent=intent))
                f_sum.classes.append(cc)
                stats['api'] += len(cc.endpoints)
        elif is_xml and (added or removed):
            if any('<changeSet' in l or 'liquibase' in l.lower() for l in added[:20]):
                stats['db'] += 1
                f_sum.db_intent = parse_liquibase(added + removed)
            else: f_sum.config_intent = "Updated XML configuration"
        elif any(p.endswith(x) for x in ['.yml', '.yaml', '.properties']) and (added or removed):
            f_sum.config_intent = extract_intent(added + removed) 
            
        release.files.append(f_sum)

    release.overview = f"Java: {stats['java']} | APIs impacted: {stats['api']} | DB Changesets: {stats['db']} | Extra API Calls: 0"
    return format_llm(release)

def format_llm(release: ReleaseSummary) -> str:
    out = [f"# Release Change Summary\n**Overview:** {release.overview}\n"]
    for f in release.files:
        if not f.classes and not f.config_intent and not f.db_intent: continue
        hdr = f"## `{f.path}` ({f.status})"
        if f.package: hdr += f" | Pkg: `{f.package}`"
        out.append(hdr)
        for c in f.classes:
            comp = f" [{c.component}]" if c.component else ""
            out.append(f"- **{c.type} {c.name}**{comp}")
            if c.endpoints: out.append(f"  - *Endpoints involved:* {', '.join(c.endpoints)}")
            for m in c.methods: out.append(f"  - **{m.name}**: {m.intent}")
        if f.db_intent: out.append(f"- **Database Schema:** {f.db_intent}")
        if f.config_intent: out.append(f"- **Config Change:** {f.config_intent}")
        out.append("")
    return "\n".join(out)

# CLI ENTRY POINT

def main():
    parser = argparse.ArgumentParser(description="Enterprise Release Diff Analyzer")
    parser.add_argument("owner", nargs='?', help="GitHub Owner")
    parser.add_argument("repo", nargs='?', help="GitHub Repo")
    parser.add_argument("base_tag", nargs='?', help="Base Tag")
    parser.add_argument("head_tag", nargs='?', help="Head Tag")
    parser.add_argument("-o", "--output", default="release_summary.md", help="Output file path")
    parser.add_argument("--generate-docs", action="store_true", help="Generate Word documentation and exit")
    
    args = parser.parse_args()
    
        
    if not all([args.owner, args.repo, args.base_tag, args.head_tag]):
        parser.error("You must provide owner, repo, base_tag, and head_tag (or use --generate-docs)")

    token = os.getenv("GITHUB_TOKEN")
    if not token: print("⚠️ Warning: GITHUB_TOKEN not set. Rate limits may apply.\n")

    try:
        diff = fetch_diff(args.owner, args.repo, args.base_tag, args.head_tag, token)
        if not diff.strip(): print("Nothing to process."); sys.exit(0)
        print("🧠 Analyzing changes (Zero-Fetch Engine)...")
        summary = generate_summary(diff)
        with open(args.output, 'w', encoding='utf-8') as file: file.write(summary)
        print(f"\n✅ Summary saved to: {args.output}")
    except Exception as e: print(f"\n❌ Error: {e}"); sys.exit(1)

if __name__ == "__main__":
    main()
