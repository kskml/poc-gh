import os
from github import Github
from tree_sitter import Language, Parser
import tree_sitter_java as tsjava

# Initialize Tree-Sitter Java
JAVA_LANGUAGE = Language(tsjava.language())
parser = Parser(JAVA_LANGUAGE)

# --- CONFIGURATION ---
GITHUB_TOKEN = "your_github_token"
REPO_NAME = "owner/repo"
BASE_TAG = "v1.0.0"
COMPARE_TAG = "v1.1.0"

def get_github_diff_info(repo, base, compare):
    """
    Fetches the comparison between two tags and returns a list of modified files 
    and their changed line ranges.
    """
    comparison = repo.compare(base, compare)
    files_data = []
    
    for file in comparison.files:
        # We store the filename and the patch (diff)
        # We will parse the patch to find changed line numbers
        files_data.append({
            "filename": file.filename,
            "patch": file.patch,
            "status": file.status
        })
    return files_data

def parse_diff_lines(patch):
    """
    Parses a git patch string to extract the line numbers that were modified.
    """
    changed_lines = set()
    if not patch:
        return changed_lines

    for line in patch.split('\n'):
        if line.startswith('@@'):
            # Format: @@ -start,len +start,len @@
            parts = line.split(' ')
            added_part = parts[2] # e.g., +123,10
            start_line = int(added_part.split(',')[0].replace('+', ''))
            # Simplified: we mark the range of the hunk as modified
            # In a production system, you'd iterate through the '+' lines specifically
            changed_lines.add(start_line) 
    return changed_lines

def get_node_text(node, source_code):
    """Extracts the actual text of a node from the source code."""
    return source_code[node.start_byte:node.end_byte].decode('utf-8')

def analyze_java_structure(source_code, changed_lines):
    """
    Uses AST to find which classes, methods, and Spring components 
    contain the changed lines.
    """
    tree = parser.parse(source_code)
    root_node = tree.root_node
    
    results = {
        "package": "",
        "modified_classes": [],
        "modified_methods": [],
        "spring_components": []
    }

    # 1. Extract Package
    pkg_node = root_node.children_get_by_type("package_declaration") # Simplified logic
    # Search for package keyword
    for child in root_node.children:
        if child.type == 'package_declaration':
            results["package"] = get_node_text(child, source_code)

    # 2. Traverse AST to find modified elements
    def traverse(node):
        # Check if this node is a Class or Method
        is_class = node.type == 'class_declaration'
        is_method = node.type == 'method_declaration'
        
        if is_class or is_method:
            # Check if any changed line falls within this node's line range
            node_start = node.start_point[0] + 1 # 0-indexed to 1-indexed
            node_end = node.end_point[0] + 1
            
            if any(line >= node_start and line <= node_end for line in changed_lines):
                text = get_node_text(node, source_code)
                
                if is_class:
                    results["modified_classes"].append(text.split('{')[0].strip())
                if is_method:
                    # Extract signature only (up to the opening brace)
                    signature = text.split('{')[0].strip()
                    results["modified_methods"].append(signature)

        # Check for Spring Annotations (looking for @RestController, @Service, etc.)
        if node.type == 'annotation':
            anno_text = get_node_text(node, source_code)
            if any(keyword in anno_text for keyword in ['@RestController', '@Service', '@Component', '@Repository', '@GetMapping', '@PostMapping']):
                # We only care if this annotation is part of a modified section
                node_line = node.start_point[0] + 1
                if any(line >= node_line - 1 and line <= node_line + 1 for line in changed_lines):
                    results["spring_components"].append(anno_text)

        for child in node.children:
            traverse(child)

    traverse(root_node)
    return results

def generate_condensed_diff_file(repo, diff_info, output_file):
    """
    Main orchestrator to process all files and write the summary.
    """
    summary_report = []

    for file_item in diff_info:
        filename = file_item['filename']
        if not filename.endswith('.java'):
            continue

        # Step 1: Get changed lines from diff
        changed_lines = parse_diff_lines(file_item['patch'])
        
        # Step 2: Fetch full file content from GitHub (Required for AST)
        content = repo.get_contents(filename, ref=COMPARE_TAG).decoded_content
        
        # Step 3: AST Analysis
        analysis = analyze_java_structure(content, changed_lines)
        
        # Step 4: Format as a summary
        file_summary = f"File: {filename}\n"
        file_summary += f"Package: {analysis['package']}\n"
        file_summary += f"Modified Classes: {', '.join(analysis['modified_classes']) if analysis['modified_classes'] else 'None'}\n"
        file_summary += f"Modified Methods: {', '.join(analysis['modified_methods']) if analysis['modified_methods'] else 'None'}\n"
        file_summary += f"Spring Components/Endpoints: {', '.join(analysis['spring_components']) if analysis['spring_components'] else 'None'}\n"
        file_summary += "-" * 50 + "\n"
        
        summary_report.append(file_summary)

    with open(output_file, "w") as f:
        f.write("### CONDENSED CODE CHANGE SUMMARY FOR LLM ###\n\n")
        f.write("\n".join(summary_report))

def main():
    g = Github(GITHUB_TOKEN)
    repo = g.get_repo(REPO_NAME)
    
    print(f"Fetching diff between {BASE_TAG} and {COMPARE_TAG}...")
    diff_info = get_github_diff_info(repo, BASE_TAG, COMPARE_TAG)
    
    print("Analyzing Java AST and condensing changes...")
    generate_condensed_diff_file(repo, diff_info, "llm_context_diff.txt")
    print("Done! Summary saved to llm_context_diff.txt")

if __name__ == "__main__":
    main()
