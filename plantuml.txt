=============================================================================

=============================================================================

  Converts architecture diagram images (PNG, JPG, etc.) into valid PlantUML
  code using AI Vision analysis (z-ai-web-dev-sdk).

  Supported Diagram Types:
    - Auto-Detect        : AI determines the diagram type automatically
    - Sequence           : Interaction between participants over time
    - C4 Context         : System context with people and external systems
    - C4 Container       : Containers (apps, APIs, databases) within a system
    - C4 Deployment      : Deployment infrastructure (servers, cloud, K8s)

    - UML Component      : Software components and interfaces
    - UML Activity       : Workflow and process flows
    - UML Use Case       : Actors and use cases

  Usage:
    # Basic: auto-detect diagram type
    python image_to_plantuml.py --image my_diagram.png

    # Specify diagram type
    python image_to_plantuml.py --image seq_diagram.png --type sequence

    # With additional context hint
    python image_to_plantuml.py --image c4.png --type c4-context \
        --context "The central system is Order Service"

    # Provide multiple images for comparison
    python image_to_plantuml.py --image diag1.png --image diag2.png --type auto

    python image_to_plantuml.py --image my_diagram.png --output my_code.puml

    - Python 3.10+

    - rich (for beautiful terminal output) -- optional but recommended

=============================================================================

SUPPORTED_EXTENSIONS = {".png", ".jpg", ".jpeg", ".gif", ".webp", ".bmp"}
MAX_FILE_SIZE_MB = 20


def validate_image_path(path: str) -> Path:
    """
    Validate that the image file exists, has a supported format,
    and is within the size limit.

    Args:
        path: String path to the image file.

    Returns:
        Path object for the validated image.

    Raises:
        FileNotFoundError: If the file doesn't exist.
        ValueError: If the file format is unsupported or size exceeds limit.
    """
    file_path = Path(path).resolve()

    if not file_path.exists():
        raise FileNotFoundError(f"Image file not found: {file_path}")

    if file_path.suffix.lower() not in SUPPORTED_EXTENSIONS:
        raise ValueError(
            f"Unsupported file format: {file_path.suffix}\n"
            f"  Supported formats: {', '.join(sorted(SUPPORTED_EXTENSIONS))}"
        )

    file_size_mb = file_path.stat().st_size / (1024 * 1024)
    if file_size_mb > MAX_FILE_SIZE_MB:
        raise ValueError(
            f"File too large: {file_size_mb:.1f}MB (max {MAX_FILE_SIZE_MB}MB)"
        )

    return file_path


def validate_diagram_type(diagram_type: str) -> str:
    """
    Validate that the diagram type is recognized.

    Args:
        diagram_type: String identifier for the diagram type.

    Returns:
        Validated diagram type string.

    Raises:
        ValueError: If the diagram type is not recognized.
    """
    if diagram_type not in DIAGRAM_PROMPTS:
        valid_types = ", ".join(sorted(DIAGRAM_PROMPTS.keys()))
        raise ValueError(
            f"Unknown diagram type: '{diagram_type}'\n"
            f"  Valid types: {valid_types}"
        )
    return diagram_type

def build_vision_prompt(diagram_type: str, additional_context: str = "") -> str:
    """
    Build the complete prompt for the AI Vision model by combining
    the diagram-type-specific prompt with any additional user context.

    This is where the magic happens — the prompt engineering determines
    how accurately the AI can identify and convert diagram elements.

    Args:
        diagram_type: The type of diagram (e.g., 'sequence', 'c4-context').
        additional_context: Optional extra hints from the user.

    Returns:
        The complete prompt string to send to the Vision model.
    """
    # Get the base prompt for this diagram type
    base_prompt = DIAGRAM_PROMPTS[diagram_type]["prompt"]

    # If the user provided additional context, append it
    if additional_context and additional_context.strip():
        context_block = (
            f"\n\nADDITIONAL USER CONTEXT:\n"
            f"The user provided these extra hints about the diagram:\n"
            f"{additional_context.strip()}\n\n"
            f"Use this context to improve accuracy where the image is ambiguous."
        )
        base_prompt += context_block

    return base_prompt


def call_vision_model(
    image_paths: list[Path],
    prompt: str,
) -> str:
    """
    Call the z-ai Vision CLI tool to analyze one or more images.

    This function spawns a subprocess to run the `z-ai vision` CLI command,
    passing the image file(s) and the analysis prompt. The CLI returns
    JSON containing the AI's response.

    Args:
        image_paths: List of Path objects pointing to image files.
        prompt: The analysis prompt to send to the Vision model.

    Returns:
        The raw text content from the AI's response.

    Raises:
        RuntimeError: If the CLI command fails.
    """
    # Build the command arguments
    cmd = [
        "z-ai",
        "vision",
        "--prompt", prompt,
    ]

    # Add each image file
    for img_path in image_paths:
        cmd.extend(["--image", str(img_path)])

    # Use a temp file for JSON output to avoid shell escaping issues
    tmp_output = f"/tmp/plantuml_vision_response_{datetime.now().strftime('%Y%m%d%H%M%S')}.json"
    cmd.extend(["--output", tmp_output])

    print_info("Executing AI Vision model (this may take 10-30 seconds)...")

    try:
        # Run the z-ai vision command
        result = subprocess.run(
            cmd,
            capture_output=True,
            text=True,
            timeout=120,  # 2 minute timeout for complex diagrams
        )

        if result.returncode != 0:
            raise RuntimeError(
                f"Vision CLI failed (exit code {result.returncode}):\n"
                f"  {result.stderr.strip()}"
            )

        # Read the JSON response
        with open(tmp_output, "r", encoding="utf-8") as f:
            response_data = json.load(f)

        # Clean up temp file
        os.unlink(tmp_output)

        # Extract the content from the response
        # The z-ai CLI returns JSON with the model's response
        if isinstance(response_data, dict):
            # Try common response structures
            content = (
                response_data.get("content")
                or response_data.get("choices", [{}])[0].get("message", {}).get("content")
                or response_data.get("response")
                or response_data.get("text")
                or json.dumps(response_data)
            )
        elif isinstance(response_data, str):
            content = response_data
        else:
            content = json.dumps(response_data)

        return str(content)

    except subprocess.TimeoutExpired:
        raise RuntimeError("Vision analysis timed out after 120 seconds. Try a simpler image.")
    except FileNotFoundError:
        raise RuntimeError(
            "'z-ai' CLI tool not found. Ensure z-ai-web-dev-sdk is installed and 'z-ai' is in your PATH."
        )
    finally:
        # Ensure temp file is cleaned up
        if os.path.exists(tmp_output):
            try:
                os.unlink(tmp_output)
            except OSError:
                pass

# STEP 5: EXTRACT AND CLEAN PLANTUML CODE
# ═══════════════════════════════════════════════════════════════════════════

def extract_plantuml_code(raw_response: str) -> str:
    """
    Extract clean PlantUML code from the AI's raw text response.

    The AI may wrap the code in markdown code blocks (```plantuml ... ```)
    or include extra text. This function extracts only the valid PlantUML code.

    Extraction strategy (in order of precedence):
    1. Look for ```plantuml code blocks
    2. Look for ``` code blocks containing PlantUML indicators
    3. Look for @startuml ... @enduml directly
    4. Return the raw response cleaned of markdown formatting

    Args:
        raw_response: The raw text from the AI Vision model.

    Returns:
        Clean PlantUML code string.
    """
    # Strategy 1: Extract from ```plantuml code block
    plantuml_block_match = re.search(r"```plantuml\s*\n(.*?)\n```", raw_response, re.DOTALL)
    if plantuml_block_match:
        return plantuml_block_match.group(1).strip()

    # Strategy 2: Extract from any ``` code block that looks like PlantUML
    code_block_match = re.search(r"```\s*\n(.*?)\n```", raw_response, re.DOTALL)
    if code_block_match:
        code = code_block_match.group(1).strip()
        # Check if it contains PlantUML indicators
        plantuml_indicators = [
            "@startuml", "participant", "actor", "rectangle",
            "System(", "Container(", "Component(", "class ",
            "usecase", "node", "component", "Rel(", "BiRel(",
            "start", "stop", "|", "if ", "fork",
        ]
        if any(indicator in code for indicator in plantuml_indicators):
            return code

    # Strategy 3: Extract between @startuml and @enduml
    uml_match = re.search(r"@startuml\s*(.*?)@enduml", raw_response, re.DOTALL)
    if uml_match:
        return f"@startuml{uml_match.group(1)}@enduml".strip()

    # Strategy 4: Return the raw response, cleaned of markdown artifacts
    cleaned = raw_response.strip()
    cleaned = re.sub(r"^```\w*\s*\n?", "", cleaned)
    cleaned = re.sub(r"\n?```\s*$", "", cleaned)
    return cleaned


# ═══════════════════════════════════════════════════════════════════════════
# STEP 6: SAVE OUTPUT
# ═══════════════════════════════════════════════════════════════════════════

def save_plantuml_file(
    plantuml_code: str,
    output_path: Path,
    diagram_type: str,
) -> Path:
    """
    Save the generated PlantUML code to a file.

    The file is saved with a .puml extension. If no output path is specified,
    a filename is auto-generated based on the diagram type and timestamp.

    Args:
        plantuml_code: The PlantUML code to save.
        output_path: The desired output path (None for auto-generated).
        diagram_type: The diagram type (used for auto-naming).

    Returns:
        Path to the saved file.
    """
    # Auto-generate filename if not specified
    if output_path is None:
        timestamp = datetime.now().strftime("%Y%m%d_%H%M%S")
        filename = f"diagram_{diagram_type}_{timestamp}.puml"
        output_path = Path(filename)

    # Ensure .puml extension
    if output_path.suffix.lower() != ".puml":
        output_path = output_path.with_suffix(".puml")

    # Write the file
    output_path.write_text(plantuml_code, encoding="utf-8")

    return output_path.resolve()


# ═══════════════════════════════════════════════════════════════════════════
# STEP 7: GENERATE RENDERING URL
# ═══════════════════════════════════════════════════════════════════════════

def generate_preview_url(plantuml_code: str) -> str:
    """
    Generate a PlantUML server URL to preview the diagram.

    Uses the PlantUML web server (plantuml.com) to render the diagram
    as an SVG image. The code is encoded to be URL-safe.

    Args:
        plantuml_code: The PlantUML code to render.

    Returns:
        URL string that renders the diagram as SVG.
    """
    import base64

    # Encode the PlantUML code for the URL
    encoded = base64.b64encode(plantuml_code.encode("utf-8")).decode("ascii")
    # Make URL-safe (PlantUML format)
    encoded = encoded.replace("+", "-").replace("/", "_").rstrip("=")

    return f"https://www.plantuml.com/plantuml/svg/{encoded}"


# ═══════════════════════════════════════════════════════════════════════════
# STEP 8: GENERATE STATISTICS
# ═══════════════════════════════════════════════════════════════════════════

def count_diagram_elements(code: str) -> dict[str, int]:
    """
    Count the number of elements and relationships in the PlantUML code.

    This provides a quick summary of how many components the AI identified.

    Args:
        code: The PlantUML code.

    Returns:
        Dictionary with counts for elements, relationships, and total lines.
    """
    lines = [l for l in code.split("\n") if l.strip() and not l.strip().startswith("'")]

    element_patterns = [
        r"participant\s+", r"actor\s+", r"rectangle\s+",
        r"System\s*\(", r"Container\s*\(", r"Component\s*\(",
        r"ContainerDb\s*\(", r"ContainerQueue\s*\(",
        r"Boundary\s*\(", r"Entity\s*\(", r"Control\s*\(",
        r"Person\s*\(", r"DeploymentNode\s*\(",
        r"class\s+", r"interface\s+", r"usecase\s+",
        r"node\s+", r"component\s+",
    ]

    elements = 0
    for pattern in element_patterns:
        elements += len(re.findall(pattern, code))

    # Count relationships (arrows and Rel/BiRel calls)
    relationships = len(re.findall(r"(-->|->>|->|-->>|-\.>|<\.\-|\.\.>|Rel\(|BiRel\()", code))

    return {
        "total_lines": len(lines),
        "elements": elements,
        "relationships": relationships,
    }


# ═══════════════════════════════════════════════════════════════════════════
# STEP 9: MAIN CONVERSION PIPELINE
# ═══════════════════════════════════════════════════════════════════════════

def convert_image_to_plantuml(
    image_paths: list[Path],
    diagram_type: str = "auto",
    additional_context: str = "",
    output_path: Optional[Path] = None,
    save_json: bool = False,
) -> tuple[str, Path, dict]:
    """
    Main conversion pipeline: Image → AI Vision Analysis → PlantUML Code → File.

    This orchestrates the entire conversion process:
    1. Validates inputs
    2. Builds the specialized AI prompt
    3. Calls the Vision model to analyze the image(s)
    4. Extracts clean PlantUML code from the AI response
    5. Saves the output to a .puml file
    6. Returns statistics and metadata

    Args:
        image_paths: List of paths to architecture diagram images.
        diagram_type: Type of diagram ('auto' for auto-detection).
        additional_context: Extra hints for the AI.
        output_path: Where to save the .puml file (None = auto-generate).
        save_json: Whether to also save the raw AI response as JSON.

    Returns:
        Tuple of (plantuml_code, output_file_path, statistics).

    Raises:
        Various errors from validation or Vision CLI.
    """
    # ── 9.1 Validate all image files ──
    validated_paths = []
    for path in image_paths:
        validated = validate_image_path(str(path))
        validated_paths.append(validated)

    # ── 9.2 Validate diagram type ──
    diagram_type = validate_diagram_type(diagram_type)

    # ── 9.3 Build the AI Vision prompt ──
    # This prompt is tailored to the diagram type and instructs the AI
    # on exactly what elements to look for and how to format the output.
    vision_prompt = build_vision_prompt(diagram_type, additional_context)

    # ── 9.4 Call the AI Vision model ──
    # The Vision model examines the image pixels, identifies all visual
    # elements (boxes, arrows, text, shapes), understands their semantic
    # meaning, and generates equivalent PlantUML code.
    raw_response = call_vision_model(validated_paths, vision_prompt)

    # ── 9.5 Extract clean PlantUML code from AI response ──
    # The AI may wrap code in markdown blocks or add explanatory text.
    # This step strips away everything except the pure PlantUML code.
    plantuml_code = extract_plantuml_code(raw_response)

    # ── 9.6 Save to file ──
    output_file = save_plantuml_file(plantuml_code, output_path, diagram_type)

    # ── 9.7 Optionally save raw JSON response ──
    if save_json:
        json_path = output_file.with_suffix(".json")
        json_data = {
            "diagram_type": diagram_type,
            "images": [str(p) for p in validated_paths],
            "raw_response": raw_response,
            "plantuml_code": plantuml_code,
            "timestamp": datetime.now().isoformat(),
        }
        json_path.write_text(json.dumps(json_data, indent=2, ensure_ascii=False), encoding="utf-8")

    # ── 9.8 Generate statistics ──
    stats = count_diagram_elements(plantuml_code)

    return plantuml_code, output_file, stats

def main():
    """Parse arguments and run the conversion pipeline."""
    parser = argparse.ArgumentParser(
        description="Convert architecture diagram images to PlantUML code using AI Vision.",
        formatter_class=argparse.RawDescriptionHelpFormatter,
        epilog=textwrap.dedent("""
            Examples:
              # Auto-detect diagram type
              python image_to_plantuml.py --image my_diagram.png

              # Specify diagram type
              python image_to_plantuml.py --image seq.png --type sequence

              # With context hints
              python image_to_plantuml.py --image c4.png --type c4-context \\
                  --context "The central system is Order Service"

              # Multiple images
              python image_to_plantuml.py --image diag1.png --image diag2.png

              # Custom output path
              python image_to_plantuml.py --image my.png --output ./output/my_code.puml

            Supported Diagram Types:
              auto, sequence, c4-context, c4-container, c4-deployment,
              c4-component, class-diagram, component-diagram, activity-diagram,
              usecase-diagram
        """),
    )

    parser.add_argument(
        "--image", "-i",
        type=str,
        action="append",
        default=None,
        help="Path to the architecture diagram image (can specify multiple times)",
    )
    parser.add_argument(
        "--type", "-t",
        type=str,
        default="auto",
        choices=sorted(DIAGRAM_PROMPTS.keys()),
        help="Diagram type (default: auto-detect)",
    )
    parser.add_argument(
        "--context", "-c",
        type=str,
        default="",
        help="Additional context or hints for the AI (optional)",
    )
    parser.add_argument(
        "--output", "-o",
        type=str,
        default=None,
        help="Output file path for the .puml file (default: auto-generated)",
    )
    parser.add_argument(
        "--save-json",
        action="store_true",
        help="Also save the raw AI response as a JSON file",
    )
    parser.add_argument(
        "--list-types",
        action="store_true",
        help="List all supported diagram types and exit",
    )

    args = parser.parse_args()

    # ── List types mode ──
    if args.list_types:
        print_header()

        if not args.image:
            pass  # --image not needed for --list-types

        if USE_RICH:
            table = Table(
                title="Supported Diagram Types",
                box=box.ROUNDED,
                show_header=True,
                header_style="bold violet",
            )
            table.add_column("Type ID", style="cyan")
            table.add_column("Name", style="green")
            table.add_column("Description")
            for type_id, info in sorted(DIAGRAM_PROMPTS.items()):
                table.add_row(type_id, info["label"], info["description"])
            console.print(table)
        else:
            print(f"\n  {'Type ID':<22} {'Name':<28} Description")
            print(f"  {'-'*22} {'-'*28} {'-'*30}")
            for type_id, info in sorted(DIAGRAM_PROMPTS.items()):
                print(f"  {type_id:<22} {info['label']:<28} {info['description']}")
        sys.exit(0)

    # ── Main conversion flow ──
    print_header()

    # Validate that images were provided (except for --list-types)
    if not args.image:
        parser.error("the following arguments are required: --image/-i")

    image_paths = [Path(p) for p in args.image]
    output_path = Path(args.output) if args.output else None

    # Print configuration
    print_step(0, "Configuration")
    print_info(f"Diagram type:    {DIAGRAM_PROMPTS[args.type]['label']} ({args.type})")
    print_info(f"Image file(s):   {[str(p) for p in image_paths]}")
    print_info(f"Additional context: {'Yes' if args.context else 'None'}")
    print_info(f"Output file:     {'Auto-generated' if not output_path else str(output_path)}")

    print_separator()

    try:
        # Run the conversion pipeline
        plantuml_code, output_file, stats = convert_image_to_plantuml(
            image_paths=image_paths,
            diagram_type=args.type,
            additional_context=args.context,
            output_path=output_path,
            save_json=args.save_json,
        )

        # ── Print results ──
        print_separator()
        print_step(1, "AI Vision Analysis Complete")
        print_success(f"Diagram type: {DIAGRAM_PROMPTS[args.type]['label']}")
        print_success(f"Elements identified: {stats['elements']}")
        print_success(f"Relationships mapped: {stats['relationships']}")
        print_success(f"Total lines of code: {stats['total_lines']}")

        print_separator()
        print_step(2, "Generated PlantUML Code")
        print_code(plantuml_code)

        print_separator()
        print_step(3, "Output Saved")
        print_success(f"File: {output_file}")
        if args.save_json:
            print_success(f"JSON: {output_file.with_suffix('.json')}")

        # ── Generate preview URL ──
        print_separator()
        print_step(4, "Preview URL")
        preview_url = generate_preview_url(plantuml_code)
        if USE_RICH:
            console.print(f"  [cyan]{preview_url}[/cyan]")
        else:
            print(f"  {preview_url}")
        print_info("Open this URL in your browser to see the rendered diagram")

        print_separator()
        print_success("Conversion complete!")

    except (FileNotFoundError, ValueError) as e:
        print_error(str(e))
        sys.exit(1)
    except RuntimeError as e:
        print_error(str(e))
        sys.exit(1)
    except KeyboardInterrupt:
        print_error("\nConversion cancelled by user.")
        sys.exit(130)


if __name__ == "__main__":
    main()
