import base64
import os
import re
from openai import OpenAI

# Initialize client
client = OpenAI(api_key=os.environ.get("OPENAI_API_KEY"))

def encode_image(image_path):
    with open(image_path, "rb") as image_file:
        return base64.b64encode(image_file.read()).decode('utf-8')

def clean_plantuml_output(raw_text):
    """
    Removes markdown blocks and conversational filler that breaks PlantUML parsers.
    """
    # Remove markdown code blocks (e.g., ```plantuml ... ```)
    # This regex looks for ```plantuml or ``` and removes it, keeping the content inside
    code = re.sub(r'```(plantuml)?', '', raw_text)
    
    # Remove common conversational openers if they exist
    # Remove anything before the first !include or @startuml
    match = re.search(r'(!include|@startuml)', code)
    if match:
        code = code[match.start():]
        
    return code.strip()

def convert_c4_image_to_plantuml(image_path, output_path="output.puml"):
    
    # The Robust Prompt from above
    system_prompt = """
    You are an expert PlantUML developer specializing in the C4 model.
    Task: Convert the provided image into valid, error-free PlantUML code.

    CRITICAL RULES TO AVOID SYNTAX ERRORS:
    1. IDENTIFY THE LEVEL FIRST:
       - If the image shows high-level boxes (People, external systems, single system box), it is a CONTEXT diagram.
       - If the image shows internal details (Web App, Database, API inside a box), it is a CONTAINER diagram.
    2. MATCH THE LIBRARY:
       - For CONTEXT: You MUST start with `!include <C4/C4_Context.puml>` and ONLY use macros: Person, System, System_Ext, Rel.
       - For CONTAINER: You MUST start with `!include <C4/C4_Container.puml>` and use macros: Person, System, System_Ext, Container, ContainerDb, Rel.
    3. SYNTAX STRICTNESS:
       - Do NOT use standard PlantUML keywords like `actor`, `rectangle`, `database`, or `package`.
       - Only use the C4 specific macros listed above.
       - Every macro must have the correct number of arguments (e.g., Rel(from, to, "label", "tech")).
       - Ensure all strings are enclosed in double quotes " ".
    4. CLEANLINESS:
       - Do NOT output markdown code blocks (```). Output the raw code only.
       - Do NOT output introductory text like "Here is the code".

    Output the final PlantUML code.
    """

    base64_image = encode_image(image_path)

    try:
        print(f"Analyzing {image_path}...")
        
        response = client.chat.completions.create(
            model="gpt-4o", 
            messages=[
                {
                    "role": "system",
                    "content": system_prompt
                },
                {
                    "role": "user",
                    "content": [
                        {"type": "text", "text": "Convert this C4 diagram to code."},
                        {
                            "type": "image_url",
                            "image_url": {
                                "url": f"data:image/png;base64,{base64_image}",
                                "detail": "high"
                            }
                        }
                    ]
                }
            ],
            temperature=0.1
        )

        raw_code = response.choices[0].message.content
        clean_code = clean_plantuml_output(raw_code)

        # Save cleaned code
        with open(output_path, "w") as f:
            f.write(clean_code)
        
        print(f"Success! Cleaned code saved to {output_path}")
        
        # Basic validation check
        if "!include" not in clean_code:
            print("WARNING: The generated code might be missing the library include.")
            
        return clean_code

    except Exception as e:
        print(f"An error occurred: {e}")
        return None

# Run it
if __name__ == "__main__":
    convert_c4_image_to_plantuml("c4_diagram.png")
