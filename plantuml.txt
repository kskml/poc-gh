import base64
import os
from openai import OpenAI
from PIL import Image
import io

# Initialize the OpenAI client
client = OpenAI(api_key=os.environ.get("OPENAI_API_KEY"))

def encode_image(image_path):
    """Loads an image and encodes it into base64 format."""
    try:
        with Image.open(image_path) as img:
            if img.mode == 'RGBA':
                img = img.convert('RGB')
            
            max_size = (1024, 1024)
            img.thumbnail(max_size)

            buffered = io.BytesIO()
            img.save(buffered, format="PNG")
            return base64.b64encode(buffered.getvalue()).decode('utf-8')
    except Exception as e:
        print(f"Error processing image: {e}")
        return None

def generate_plantuml_from_image(image_path):
    """
    Sends the image to GPT-4o with a FIXED system prompt to handle C4 libraries correctly.
    """
    base64_image = encode_image(image_path)
    if not base64_image:
        return None

    # --- UPDATED SYSTEM PROMPT ---
    system_prompt = """
    You are an expert Software Architect and a PlantUML generator. Your goal is to analyze the uploaded architecture image and convert it into valid, error-free PlantUML code.

    Follow these rules strictly:
    1. Identify the diagram type:
       - If it is a Sequence Diagram, use 'participant', '->', 'activate', 'deactivate'.
       - If it is a C4 Model (Context or Container), use the REMOTE PUBLIC URL includes. 
         DO NOT use local includes (e.g., '!include C4_Container.puml') because the user does not have these files.
         
         USE THESE URLS INSTEAD:
         - Context: '!include https://raw.githubusercontent.com/plantuml-stdlib/C4-PlantUML/master/C4_Context.puml'
         - Container: '!include https://raw.githubusercontent.com/plantuml-stdlib/C4-PlantUML/master/C4_Container.puml'
         
       - If it is a generic Architecture/Deployment diagram, use standard rectangles, nodes, and cloud components.

    2. Visual Estimation:
       - Infer logical names if text is blurry (e.g., "Database", "Auth Service", "Client App").
       - Preserve spatial hierarchy.

    3. Output Format:
       - Provide ONLY the raw PlantUML code starting with @startuml and ending with @enduml.
       - Do NOT provide markdown formatting (like ```plantuml).
       - Do NOT include introductory text.

    4. Syntax Validity:
       - Ensure all arrows and labels are valid.
    """

    try:
        response = client.chat.completions.create(
            model="gpt-4o",
            messages=[
                {
                    "role": "system",
                    "content": system_prompt
                },
                {
                    "role": "user",
                    "content": [
                        {
                            "type": "text",
                            "text": "Convert this architecture diagram into valid PlantUML code."
                        },
                        {
                            "type": "image_url",
                            "image_url": {
                                "url": f"data:image/png;base64,{base64_image}",
                                "detail": "high"
                            }
                        }
                    ]
                }
            ],
            temperature=0.2,
            max_tokens=2000
        )

        content = response.choices[0].message.content
        
        # Cleanup: Remove markdown code blocks if present
        if "```plantuml" in content:
            content = content.replace("```plantuml", "").replace("```", "")
        
        return content.strip()

    except Exception as e:
        print(f"API Request failed: {e}")
        return None

def save_plantuml_file(plantuml_code, output_filename="diagram.puml"):
    with open(output_filename, "w") as f:
        f.write(plantuml_code)
    print(f"Successfully saved PlantUML code to {output_filename}")

if __name__ == "__main__":
    input_image = "architecture.png"
    
    if os.path.exists(input_image):
        print(f"Processing {input_image}...")
        uml_code = generate_plantuml_from_image(input_image)
        
        if uml_code:
            save_plantuml_file(uml_code)
            print("\n--- Generated PlantUML Code ---")
            print(uml_code)
        else:
            print("Failed to generate code.")
    else:
        print(f"File {input_image} not found.")
