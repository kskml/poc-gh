import base64
import os
from openai import OpenAI
from PIL import Image
import io

# Initialize the OpenAI client (make sure OPENAI_API_KEY is set in your environment variables)
client = OpenAI(api_key=os.environ.get("OPENAI_API_KEY"))

def encode_image(image_path):
    """
    Loads an image and encodes it into base64 format required by the OpenAI API.
    """
    try:
        # Open image to ensure it exists and is valid, also compressing if necessary
        with Image.open(image_path) as img:
            # Convert to RGB if RGBA (to avoid PNG transparency issues in some viewers)
            if img.mode == 'RGBA':
                img = img.convert('RGB')
            
            # Resize if too large to save tokens (OpenAI limit is usually 20MB, but smaller is faster)
            max_size = (1024, 1024)
            img.thumbnail(max_size)

            buffered = io.BytesIO()
            img.save(buffered, format="PNG")
            return base64.b64encode(buffered.getvalue()).decode('utf-8')
    except Exception as e:
        print(f"Error processing image: {e}")
        return None

def generate_plantuml_from_image(image_path):
    """
    Sends the image to GPT-4o with the system prompt and returns the PlantUML code.
    """
    base64_image = encode_image(image_path)
    if not base64_image:
        return None

    # The System Prompt defined in the previous section
    system_prompt = """
    You are an expert Software Architect and a PlantUML generator. Your goal is to analyze the uploaded architecture image and convert it into valid, error-free PlantUML code.

    Follow these rules strictly:
    1. Identify the diagram type:
       - If it is a Sequence Diagram, use 'participant', '->', 'activate', 'deactivate'.
       - If it is a C4 Model (Context or Container), use '!include C4_Context.puml' or '!include C4_Container.puml', 'System', 'Person', 'Rel'.
       - If it is a generic Architecture/Deployment diagram, use standard rectangles, nodes, and cloud components.

    2. Visual Estimation:
       - Since text inside the image might be blurry or small, infer the logical names of components based on their position and context.
       - Preserve the spatial hierarchy: if Box A is inside Box B, nest them in the PlantUML code.

    3. Output Format:
       - Provide ONLY the raw PlantUML code starting with @startuml and ending with @enduml.
       - Do NOT provide markdown formatting (like ```plantuml).
       - Do NOT include introductory text or explanations. Just the code.

    4. Syntax Validity:
       - Ensure all relationships have valid arrow syntax.
       - Ensure all labels are closed properly.
    """

    try:
        response = client.chat.completions.create(
            model="gpt-4o", # GPT-4o is recommended for best vision capabilities
            messages=[
                {
                    "role": "system",
                    "content": system_prompt
                },
                {
                    "role": "user",
                    "content": [
                        {
                            "type": "text",
                            "text": "Convert this architecture diagram into valid PlantUML code."
                        },
                        {
                            "type": "image_url",
                            "image_url": {
                                "url": f"data:image/png;base64,{base64_image}",
                                "detail": "high" # Use "high" for OCR and diagram details
                            }
                        }
                    ]
                }
            ],
            temperature=0.2, # Low temperature for deterministic code generation
            max_tokens=2000
        )

        content = response.choices[0].message.content
        
        # Cleanup: Remove markdown code blocks if the model ignored instructions
        if "```plantuml" in content:
            content = content.replace("```plantuml", "").replace("```", "")
        
        return content.strip()

    except Exception as e:
        print(f"API Request failed: {e}")
        return None

def save_plantuml_file(plantuml_code, output_filename="diagram.puml"):
    with open(output_filename, "w") as f:
        f.write(plantuml_code)
    print(f"Successfully saved PlantUML code to {output_filename}")

if __name__ == "__main__":
    # Update this path to your image file
    input_image = "architecture.png"
    
    if os.path.exists(input_image):
        print(f"Processing {input_image}...")
        uml_code = generate_plantuml_from_image(input_image)
        
        if uml_code:
            save_plantuml_file(uml_code)
            print("\n--- Generated PlantUML Code ---")
            print(uml_code)
        else:
            print("Failed to generate code.")
    else:
        print(f"File {input_image} not found. Please place a png file in the directory.")
