import base64
import os
import re
from openai import OpenAI

# Initialize client
client = OpenAI(api_key=os.environ.get("OPENAI_API_KEY"))

def encode_image(image_path):
    with open(image_path, "rb") as image_file:
        return base64.b64encode(image_file.read()).decode('utf-8')

def clean_output(text):
    # Remove markdown blocks if present
    text = re.sub(r'```(plantuml)?', '', text)
    # Remove text before the first !include or @startuml
    match = re.search(r'(!include|@startuml)', text)
    if match:
        text = text[match.start():]
    return text.strip()

def generate_c4_code(image_path, output_path="output.puml"):
    
    # The Force-Content Prompt
    system_prompt = """
    You are an expert Software Architect and PlantUML generator.
    Task: Convert the C4 architecture image into fully populated PlantUML code.

    CRITICAL: You MUST populate the body of the diagram. Do not output only headers/footers.

    RULES:
    1. Library Selection:
       - If the diagram shows high-level concepts (User -> System), use: !include <C4/C4_Context.puml>
       - If the diagram shows internal components (Web App, Database), use: !include <C4/C4_Container.puml>
    2. Element Extraction:
       - Identify EVERY box in the image.
       - If you cannot read the text clearly on a box, assign it a generic name like "System_A", "Component_B", or "Database_1".
       - You MUST NOT skip any visible element.
    3. Relationship Extraction:
       - Identify EVERY arrow connecting the boxes.
       - Use the Rel macro: Rel(Source, Target, "Label", "Tech").
       - If the label is unreadable, use "Uses" or "Interacts".
    4. Syntax:
       - Use standard C4 macros: Person, System, System_Ext, Container, ContainerDb, Rel.
       - Do NOT use standard PlantUML shapes (rectangle, actor).

    Output:
    Output the complete PlantUML code starting with @startuml and ending with @enduml. Ensure there is code between the start and end tags.
    """

    base64_image = encode_image(image_path)

    try:
        print(f"Generating code for {image_path}...")
        
        response = client.chat.completions.create(
            model="gpt-4o",
            messages=[
                {"role": "system", "content": system_prompt},
                {
                    "role": "user",
                    "content": [
                        {"type": "text", "text": "Analyze this C4 diagram and write the full PlantUML code."},
                        {
                            "type": "image_url",
                            "image_url": {
                                "url": f"data:image/png;base64,{base64_image}",
                                "detail": "high"
                            }
                        }
                    ]
                }
            ],
            temperature=0.1,
            # Explicitly setting max_tokens to ensure it has space to write the definitions
            max_tokens=2048 
        )

        raw_code = response.choices[0].message.content
        final_code = clean_output(raw_code)

        # Save the file
        with open(output_path, "w") as f:
            f.write(final_code)
            
        # Validation Check
        # We check if the file is suspiciously short (less than 150 chars usually means just headers)
        if len(final_code) < 150:
            print("WARNING: The generated code is very short. The model might have missed the content.")
            print("Generated content:", final_code)
        else:
            print(f"Success! Code saved to {output_path}")

        return final_code

    except Exception as e:
        print(f"Error: {e}")
        return None

if __name__ == "__main__":
    generate_c4_code("c4_diagram.png")
